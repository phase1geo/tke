52  # Useful process for debugging.
53  proc stacktrace {} {
54 
55  set stack "Stack trace:\n"
56  for {set i 1} {$i < [info level]} {incr i} {
57  set lvl [info level -$i]
58  set pname [lindex $lvl 0]
59  if {[namespace which -command $pname] eq ""} {
60  for {set j [expr $i + 1]} {$j < [info level]} {incr j} {
61  if {[namespace which -command [lindex [info level -$j] 0]] ne ""} {
62  set pname "[namespace qualifiers [lindex [info level -$j] 0]]::$pname"
63  break
64  }
65  }
66  }
67  append stack [string repeat " " $i]$pname
68  foreach value [lrange $lvl 1 end] arg [info args $pname] {
69  if {$value eq ""} {
70  info default $pname $arg value
71  }
72  append stack " $arg='$value'"
73  }
74  append stack \n
75  }
76 
77  return $stack
78 
79  }

218  # Converts an RGB value into an HSV value.
219  proc rgb_to_hsv {r g b} {
220 
221  set sorted [lsort -real [list $r $g $b]]
222  set temp [lindex $sorted 0]
223  set v [lindex $sorted 2]
224 
225  set bottom [expr {$v-$temp}]
226  if {$bottom == 0} {
227  set h 0
228  set s 0
229  set v $v
230  } else {
231  if {$v == $r} {
232  set top [expr {$g-$b}]
233  if {$g >= $b} {
234  set angle 0
235  } else {
236  set angle 360
237  }
238  } elseif {$v == $g} {
239  set top [expr {$b-$r}]
240  set angle 120
241  } elseif {$v == $b} {
242  set top [expr {$r-$g}]
243  set angle 240
244  }
245  set h [expr { round( 60 * ( double($top) / $bottom ) + $angle ) }]
246  }
247 
248  if {$v == 0} {
249  set s 0
250  } else {
251  set s [expr { round( 255 - 255 * ( double($temp) / $v ) ) }]
252  }
253 
254  return [list $h $s $v]
255 
256  }

259  # Converts an HSV value into an RGB value.
260  proc hsv_to_rgb {h s v} {
261 
262  set hi [expr { int( double($h) / 60 ) % 6 }]
263  set f [expr { double($h) / 60 - $hi }]
264  set s [expr { double($s)/255 }]
265  set v [expr { double($v)/255 }]
266  set p [expr { double($v) * (1 - $s) }]
267  set q [expr { double($v) * (1 - $f * $s) }]
268  set t [expr { double($v) * (1 - (1 - $f) * $s) }]
269 
270  switch -- $hi {
271  0 {
272  set r $v
273  set g $t
274  set b $p
275  }
276  1 {
277  set r $q
278  set g $v
279  set b $p
280  }
281  2 {
282  set r $p
283  set g $v
284  set b $t
285  }
286  3 {
287  set r $p
288  set g $q
289  set b $v
290  }
291  4 {
292  set r $t
293  set g $p
294  set b $v
295  }
296  5 {
297  set r $v
298  set g $p
299  set b $q
300  }
301  default {
302  error "Wrong hi value in hsv_to_rgb procedure! This should never happen!"
303  }
304  }
305 
306  set r [expr {round($r*255)}]
307  set g [expr {round($g*255)}]
308  set b [expr {round($b*255)}]
309 
310  return [list $r $g $b]
311 
312  }

315  # Automatically adjusts the given color by a value equal to diff such
316  # that if color is a darker color, the value will be lightened and if
317  # color is a lighter color, the value will be darkened.
318  proc auto_adjust_color {color diff} {
319 
320  # Create the lighter version of the primary color
321  lassign [winfo rgb . $color] r g b
322  lassign [rgb_to_hsv [expr $r >> 8] [expr $g >> 8] [expr $b >> 8]] hue saturation value
323  set value [expr ($value < 128) ? ($value + $diff) : ($value - $diff)]
324  set rgb [hsv_to_rgb $hue $saturation $value]
325 
326  return [format {#%02x%02x%02x} {*}$rgb]
327 
328  }
