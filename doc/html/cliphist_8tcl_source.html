33  # Load the contents of the saved clipboard history.
34  proc load {} {
35 
36  variable cliphist_file
37  variable hist
38 
39  if {![catch { open $cliphist_file r} rc]} {
40 
41  # Read the contents of the file
42  set contents [read $rc]
43  close $rc
44 
45  # Parse the file
46  set in_history 0
47  foreach line [split $contents \n] {
48  if {$line eq "clipping:"} {
49  set in_history 1
50  set clipping ""
51  } elseif {$in_history && [regexp {^\t(.*)$} $line -> str]} {
52  append clipping "$str\n"
53  } elseif {$in_history} {
54  set clipping [string range $clipping 0 end-1]
55  set in_history 0
56  lappend hist $clipping
57  }
58  }
59 
60  }
61 
62  # Handle any changes to the clipboard history depth
63  trace variable preferences::prefs(Tools/ClipboardHistoryDepth) w cliphist::handle_cliphist_depth
64 
65  }

68  # Saves the state of the clipboard history to a file. This is called
69  # prior to exiting the application.
70  proc save {} {
71 
72  variable cliphist_file
73  variable hist
74 
75  if {![catch { open $cliphist_file w} rc]} {
76  foreach clipping $hist {
77  puts $rc "clipping:"
78  foreach line [split $clipping \n] {
79  puts $rc "\t$line"
80  }
81  puts $rc ""
82  }
83  close $rc
84  }

89  # Adds the current clipboard contents to the clipboard history list.
90  proc add_from_clipboard {} {
91 
92  variable hist
93 
94  # Get the clipboard content
95  set str [string map {\{ \\\{} [clipboard get]]
96 
97  if {[string trim $str] ne ""} {
98 
99  # If the string doesn't exist in history, add it
100  if {[set index [lsearch -exact $hist $str]] == -1} {
101 
102  # Append the string to the history file
103  lappend hist $str
104 
105  # Trim the history to meet the maxsize requirement, if necessary
106  if {[llength $hist] > [preferences::get Tools/ClipboardHistoryDepth]} {
107  set hist [lrange $hist 1 end]
108  }
109 
110  # Otherwise, move the current string to the beginning of the history
111  } else {
112 
113  set hist [linsert [lreplace $hist $index $index] end $str]
114 
115  }
116 
117  }
118 
119  }

130  # Adds the current string from the clipboard history list to the clipboard

133  proc add_to_clipboard {str} {
134

145  # Returns the clipboard history as a list of string pairs where the
146  # first item is the value to use in the listbox while the second pair
147  # should be used in the full detail.
148  proc get_history {} {
149 
150  variable hist
151 
152  set items [list]
153 
154  foreach item [lreverse $hist] {
155  set lines [split $item \n]
156  set short [lindex $lines 0]
157  if {[llength $lines] > 1} {
158  append short " ..."
159  }
160  lappend items [list [string map {\\\{ \{} $short] [string map {\\\{ \{} $item]]
161  }
162 
163  return $items
164 
165  }
