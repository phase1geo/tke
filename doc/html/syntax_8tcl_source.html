126  # Adds the given syntax file to the total list.
127  proc add_syntax {sfile {interp ""}} {
128 
129  variable langs
130  variable lang_template
131  variable filetypes
132 
133  # Get the name of the syntax
134  set name [file rootname [file tail $sfile]]
135 
136  # Initialize the language array
137  array set lang_array [array get lang_template]
138 
139  # Read the file
140  if {![catch { open $sfile r} rc]} {
141 
142  # Read in the file information
143  array set lang_array [read $rc]
144  close $rc
145 
146  # Format the extension information
147  set extensions [list]
148  foreach pattern $lang_array(filepatterns) {
149  if {[regexp {^\.\w+$} [set extension [file extension $pattern]]]} {
150  lappend extensions $extension
151  }
152  }
153  set lang_array(extensions) $extensions
154  if {[llength $extensions] > 0} {
155  lappend filetypes [list "$name Files" $extensions TEXT]
156  }
157 
158  # Sort the filetypes by name
159  set filetypes [lsort -index 0 $filetypes]
160 
161  # Add the interpreter
162  set lang_array(interp) $interp
163 
164  # Add the language and the command launcher
165  set langs($name) [array get lang_array]
166  launcher::register [format "%s: %s" [msgcat::mc "Syntax"] $name] [list syntax::set_current_language $name]
167 
168  }
169 
170  }

231  # Given the specified filename, returns the language name that supports
232  # it. If multiple languages respond, use the first match.
233  proc get_default_language {filename} {
234 
235  variable langs
236  variable assoc_file
237 
238  # Check to see if the user has specified a language override for files like
239  # the filename.
240  if {![catch { tkedat::read $assoc_file 0} rc]} {
241  array set associations $rc
242  set key [file dirname $filename],[file extension $filename]
243  if {[info exists associations($key)]} {
244  return $associations($key)
245  }
246  }
247 
248  # Get the list of extension overrides
249  array set overrides [preferences::get {General/LanguagePatternOverrides}]
250 
251  set maxlen 0
252  set best_match [msgcat::mc "None"]
253 
254  foreach lang [array names langs] {
255  array set lang_array $langs($lang)
256  set patterns $lang_array(filepatterns)
257  set excluded 0
258  if {[info exists overrides($lang)]} {
259  set exclude_patterns [list]
260  foreach pattern $overrides($lang) {
261  switch [string index $pattern 0] {
262  "+" { lappend patterns [string range $pattern 1 end]}
263  "-" { lappend epatterns [string range $pattern 1 end]}
264  }
265  }
266  foreach pattern $epatterns {
267  if {[string match -nocase $pattern [file tail $filename]]} {
268  set excluded 1
269  break
270  }
271  }
272  }
273  if {!$excluded} {
274  foreach pattern $patterns {
275  if {[string match -nocase $pattern $filename]} {
276  if {[string length $pattern] > $maxlen} {
277  set maxlen [string length $pattern]
278  set best_match $lang
279  }
280  }
281  }
282  }
283  }
284 
285  return $best_match
286 
287  }

394 
395  variable langs
396  variable curr_lang
397  variable current_lang
398  variable meta_tags
399 
400  array set opts {
401  -highlight 1
402  }
403  array set opts $args
404 
405  # Get the current syntax theme
406  array set theme [theme::get_syntax_colors]
407 
408  # Clear the syntax highlighting for the widget
409  if {$opts(-highlight)} {
410  ctext::clearHighlightClasses $txt
411  ctext::setBlockCommentPatterns $txt {} {}
412  ctext::setLineCommentPatterns $txt {} {}
413  ctext::setStringPatterns $txt {} {}
414  ctext::setAutoMatchChars $txt {} {}
415  }
416 
417  # Set default indent/unindent strings
418  indent::set_indent_expressions $txt.t {} {} {}
419 
420  # Apply the new syntax highlighting syntax, if one exists for the given language
421  if {[info exists langs($language)]} {
422 
423  if {[catch {
424 
425  array set lang_array $langs($language)
426 
427  # Get the command prefix and create a namespace for the language, if necessary
428  if {$lang_array(interp) ne ""} {
429  set cmd_prefix "$lang_array(interp) eval "
430  set lang_ns ""
431  $lang_array(interp) alias $txt $txt
432  } else {
433  set cmd_prefix ""
434  set lang_ns [string tolower $language]
435  }
436 
437  # Initialize the meta tags
438  set meta_tags($txt) [list meta readmeta]
439 
440  # Set the case sensitivity, delimiter characters and wrap mode
441  $txt configure -casesensitive $lang_array(casesensitive) -escapes $lang_array(escapes)

443  $txt configure -delimiters $lang_array(delimiters)
444  }
445

448  syntax { $txt configure -wrap [expr {$lang_array(linewrap) ? "word" : "none"}]}
449  enable { $txt configure -wrap "word"}
450  disable { $txt configure -wrap "none"}
451  }
452

454  ctext::addHighlightClass $txt keywords $theme(keywords)
455  ctext::addHighlightKeywords $txt $lang_array(keywords) class keywords
456 
457  # Add the rest of the sections
458  set_language_section $txt symbols $lang_array(symbols) "" $cmd_prefix $lang_ns
459  set_language_section $txt punctuation $lang_array(punctuation) ""
460  set_language_section $txt numbers $lang_array(numbers) ""
461  set_language_section $txt precompile $lang_array(precompile) ""
462  set_language_section $txt miscellaneous1 $lang_array(miscellaneous1) ""
463  set_language_section $txt miscellaneous2 $lang_array(miscellaneous2) ""
464  set_language_section $txt miscellaneous3 $lang_array(miscellaneous3) ""
465  set_language_section $txt highlighter $lang_array(highlighter) ""
466  set_language_section $txt meta $lang_array(meta) ""
467  set_language_section $txt readmeta $lang_array(readmeta) ""
468  set_language_section $txt advanced $lang_array(advanced) "" $cmd_prefix $lang_ns
469 
470  # Add the comments, strings and indentations
471  ctext::clearCommentStringPatterns $txt
472  ctext::setBlockCommentPatterns $txt {} $lang_array(bcomments) $theme(comments)
473  ctext::setLineCommentPatterns $txt {} $lang_array(lcomments) $theme(comments)
474  ctext::setStringPatterns $txt {} $lang_array(strings) $theme(strings)
475  ctext::setIndentation $txt {} $lang_array(indent) indent
476  ctext::setIndentation $txt {} $lang_array(unindent) unindent
477 
478  set reindentStarts [list]
479  set reindents [list]
480  foreach reindent $lang_array(reindent) {
481  lappend reindentStarts [lindex $reindent 0]
482  lappend reindents {*}[lrange $reindent 1 end]
483  }
484  ctext::setIndentation $txt {} $reindentStarts reindentStart
485  ctext::setIndentation $txt {} $reindents reindent
486 
487  # Add the FIXME
488  ctext::addHighlightClass $txt fixme $theme(miscellaneous1)
489  ctext::addHighlightKeywords $txt FIXME class fixme
490 
491  # Set the indent/unindent regular expressions
492  indent::set_indent_expressions $txt.t $lang_array(indent) $lang_array(unindent) $lang_array(reindent) 0
493 
494  # Set the completer options for the given language
495  ctext::setAutoMatchChars $txt {} $lang_array(matchcharsallowed)
496  completer::set_auto_match_chars $txt.t {} $lang_array(matchcharsallowed)

504  add_sublanguage $txt $sublang $cmd_prefix "" {} {}
505  }
506  }
507 
508  # Set the snippets for the current text widget
509  snippets::set_language $language
510  snippets::set_expandtabs $txt [expr $lang_array(tabsallowed) ? 0 : 1]
511 
512  } rc]} {
513  gui::set_error_message [format "%s (%s)" [msgcat::mc "Syntax error in syntax file"] $language] $rc
514  puts $::errorInfo
515  }
516 
517  }
518 
519  # Save the language
520  set curr_lang($txt) [set current_lang $language]
521 
522  # Re-highlight
523  if {$opts(-highlight)} {
524  $txt highlight 1.0 end

526  }

618  # Adds syntax highlighting for a given type
619  proc set_language_section {txt section section_list lang {cmd_prefix ""} {lang_ns ""}} {
620 
621  variable theme
622  variable meta_tags
623 
624  # Get the current syntax theme
625  array set theme [theme::get_syntax_colors]
626 
627  switch $section {
628  "advanced" -
629  "symbols" {
630  while {[llength $section_list]} {
631  set section_list [lassign $section_list type]
632  switch -glob $type {
633  "HighlightClass" {
634  if {$section eq "advanced"} {
635  set section_list [lassign $section_list name color modifiers]
636  switch $color {
637  none { ctext::addHighlightClass $txt $name "" "" $modifiers}
638  highlighter { ctext::addHighlightClass $txt $name $theme(background) $theme($color) $modifiers}
639  default { ctext::addHighlightClass $txt $name $theme($color) "" $modifiers}
640  }
641  if {[lsearch {meta readmeta} $color] != -1} {
642  lappend meta_tags($txt) $name
643  $txt tag raise _$name
644  }
645  }
646  }
647  "HighlightProc" {
648  if {$section eq "advanced"} {
649  set section_list [lassign $section_list name body]
650  if {([llength $section_list] > 0) && ![string match Highlight* [lindex $section_list 0]]} {
651  set params $body
652  set section_list [lassign $section_list body]
653  } else {
654  set params [list txt startpos endpos ins]
655  }
656  if {$lang_ns ne ""} {
657  namespace eval $lang_ns [list proc $name $params [subst -nocommands -novariables $body]]
658  }
659  }
660  }
661  "HighlightEndProc" -
662  "TclEnd" -
663  "IgnoreEnd" {
664  # This is not invalid syntax but is not used for anything in this namespace
665  }
666  "Highlight*" {
667  set section_list [lassign $section_list syntax command]
668  if {$command ne ""} {
669  if {$cmd_prefix ne ""} {
670  ctext::add$type $txt $syntax command "$cmd_prefix $command" $lang
671  } elseif {[string first :: $command] != -1} {
672  ctext::add$type $txt $syntax command $command $lang
673  } else {
674  ctext::add$type $txt $syntax command syntax::${lang_ns}::$command $lang
675  }
676  } else {
677  ctext::add$type $txt $syntax class [expr {($section eq "symbols") ? "symbols" : "none"}] $lang
678  }
679  }
680  "TclBegin" {
681  set section_list [lassign $section_list content]
682  namespace eval $lang_ns $content
683  }
684  "IgnoreBegin" {
685  set section_list [lrange $section_list 1 end]
686  }
687  default {
688  return -code error "Syntax error found in $section section -- bad type $type"
689  }
690  }
691  }
692  }
693  "highlighter" {
694  foreach {type syntax modifiers} $section_list {
695  if {$syntax ne ""} {
696  set class $section
697  if {[llength $modifiers] > 0} {
698  append class -[join $modifiers -]
699  }
700  ctext::addHighlightClass $txt $class $theme(background) $theme($section) $modifiers
701  ctext::add$type $txt $syntax class $class $lang
702  }
703  }
704  }
705  default {
706  foreach {type syntax modifiers} $section_list {
707  if {$syntax ne ""} {
708  set class $section
709  if {[llength $modifiers] > 0} {
710  append class -[join $modifiers -]
711  }
712  ctext::addHighlightClass $txt $class $theme($section) "" $modifiers
713  ctext::add$type $txt $syntax class $class $lang
714  }
715  }
716  }
717  }
718 
719  }

722  # Returns true if the given text widget contains meta characters.
723  proc contains_meta_chars {txt} {
724 
725  variable meta_tags
726 
727  set all_tags [ctext::getHighlightClasses $txt]
728
