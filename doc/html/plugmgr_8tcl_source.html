57  # Adds a single plugin to the plugin database file. Returns the
58  # data that is stored in the plugin entry.
59  proc add_plugin {dbfile name args} {
60 
61  variable default_pdata
62 
63  # Store the important plugin values
64  array set pdata [array get default_pdata]
65  foreach {attr value} $args {
66  if {[info exists pdata($attr)]} {
67  set pdata($attr) $value
68  }
69  }
70 
71  if {[file exists $dbfile]} {
72 
73  # Read in the existing values
74  if {[catch { tkedat::read $dbfile} rc]} {
75  return -code error "Unable to read the given plugin database file"
76  }
77 
78  array set data $rc
79  array set plugins $data(plugins)
80 
81  }
82 
83  set plugins($name) [array get pdata]
84  set data(plugins) [array get plugins]
85 
86  # Save the file
87  if {[catch { tkedat::write $dbfile [array get data]} rc]} {
88  return -code error "Unable to update the plugin database file"
89  }
90 
91  return [array get pdata]
92 
93  }

221  # Get version information from the given plugin directory.
222  proc get_versions {plugdir} {
223 
224  if {[catch { tkedat::read [file join $plugdir header.tkedat]} rc]} {
225  return [list "1.0" {}]
226  } else {
227  array set header $rc
228  set i 0
229  set version [split $header(version) .]
230  foreach num $version {
231  if {$i == 0} {
232  lappend next_versions "[expr $num + 1].0"
233  } else {
234  lappend next_versions [join [list {*}[lrange $version 0 [expr $i - 1]] [expr $num + 1]] .]
235  }
236  incr i
237  }
238  set next_versions [linsert $next_versions end-1 [join [list {*}$version 1] .]]
239  return [list $header(version) [lreverse $next_versions]]
240  }
241 
242  }

262

270

272  set contents(version) $version

274  return -code error "Unable to write header.tkedat"

938  # Loads the plugin database file from the server.
939  proc load_database {} {
940 
941  variable database
942 
943  set url "http://tke.sourceforge.net/plugins/plugins.tkedat"
944 
945  # Download the database to a local file
946  if {[set fname [utils::download_url $url]] eq ""} {
947  show_error_message [msgcat::mc "Unable to fetch plugin database"]
948  return
949  }
950 
951  # Load the downloaded file
952  if {[catch { tkedat::read $fname} rc]} {
953  file delete -force $fname
954  show_error_message [msgcat::mc "Unable to load plugin database file"]
955  return
956  }
957 
958  catch { file delete -force $fname}
959 
960  # Make sure that the database is cleared
961  array unset database
962 
963  array set database $rc
964  array set new_plugins $database(plugins)
965 
966  # Initialize the new plugins database
967  foreach name [array names new_plugins] {
968  array set new_plugin $new_plugins($name)
969  set new_plugin(update_avail) 0
970  set new_plugin(installed) 0
971  set new_plugins($name) [array get new_plugin]
972  }
973 
974  # Load the local file and compare the old versions to the new versions
975  if {![catch { tkedata::read [file join $::tke_home iplugins plugins.tkedat]} rc]} {
976 
977  array set old_data $rc
978  array set old_plugins $old_data(plugins)
979 
980  # Cross-reference the old database with the new, updating the new
981  foreach name [array names old_plugins] {
982  if {[info exists new_plugins($name)]} {
983  array set old_plugin $old_plugins($name)
984  array set new_plugin $new_plugins($name)
985  set new_plugin(update_avail) [expr {$old_plugin(version) ne $new_plugin(version)}]
986  set new_plugin(installed) 1
987  set new_plugins($name) [array get new_plugin]
988  }
989  }
990 
991  }
992 
993  # Save the new plugins data back to the database
994  set database(plugins) [array get new_plugins]
995 
996  }
