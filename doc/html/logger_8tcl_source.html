34  # Called at application start to initialize the debug logfile.
35  proc initialize {} {
36 
37  variable logdir
38  variable logrc
39 
40  # Get the logfile directory
41  if {[set logdir [[ns preferences]::get Debug/LogDirectory]] eq ""} {
42  set logdir [file join $::tke_home logs]
43  }
44 
45  # Get the native logfile name and create the directory if it does not already exist
46  if {![file exists [set logdir [file normalize $logdir]]]} {
47  file mkdir $logdir
48  }
49 
50  # Create the logfile
51  create_logfile $logdir
52 
53  # Keep an eye on the Debug/LogDirectory preference option
54  trace variable [ns preferences]::prefs(Debug/LogDirectory) w logger::handle_logdir_change
55 
56  }

97  # Handles any changes to the Debug/LogDirectory preference option.
98  proc handle_logdir_change {name1 name2 op} {
99 
100  variable logdir
101 
102  # Get the preference directory value
103  if {[set pref_dir [[ns preferences]::get Debug/LogDirectory]] eq ""} {
104  set pref_dir [file join $::tke_home logs]
105  }
106 
107  # Normalize the preference directory
108  set pref_dir [file normalize $pref_dir]
109 
110  # If the directory exists and it differs from the original, close, move and re-open the logfile
111  if {$logdir ne $pref_dir} {
112 
113  # Create the directory if it does not exist
114  if {![file exists $pref_dir]} {
115  file mkdir $pref_dir
116  }
117 
118  # If the logfile was previously opened, close it, move it and re-open for appendment
119  if {$logrc ne ""} {
120 
121  # Close the logfile
122  close $logrc
123 
124  # Move the logfile
125  file rename -force [file join $logdir debug.[pid].log] $pref_dir
126 
127  # Reopen the logfile
128  if {![catch { open [file join $pref_dir debug.[pid].log] a} rc]} {
129  fconfigure $rc -buffering line
130  set logrc $rc
131  }
132 
133  # Set the logfile directory name to the preference name
134  set logdir $pref_dir
135 
136  # Otherwise, open the logfile in the new directory for writing
137  } else {
138 
139  # Create the logfile
140  create_logfile $pref_dir
141 
142  }
143 
144  }
145 
146  }

151  proc log {str} {
152 
153  variable logrc
