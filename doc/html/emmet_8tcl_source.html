58  # Returns a three element list containing the snippet text, starting and ending
59  # position of that text.
60  proc get_snippet_text_pos {tid} {
61 
62  variable data
63 
64  # Get the current text widget
65  set txt [[ns gui]::current_txt $tid]
66 
67  # Get the current line and column numbers
68  lassign [split [$txt index insert] .] line endcol
69 
70  # Get the current line
71  set str [$txt get "insert linestart" insert]
72 
73  # Get the prespace of the current line
74  regexp {^([ \t]*)} $str -> prespace
75 
76  # If we have a tag, ignore all text prior to it
77  set startcol [expr {[regexp $data(tag) $str match] ? [string length $match] : 0}]
78 
79  # Gather the positions of any square or curly brackets in the left-over area
80  foreach key [list brackets space] {
81  set pos($key) [list]
82  set col $startcol
83  while {[regexp -start $col -- $data($key) $str match pre]} {
84  lappend pos($key) [expr $col + [string length $pre]] [expr $col + [string length $match]]
85  set col [lindex $pos($key) end]
86  }
87  }
88 
89  # See if there is a space which does not exist within a square or curly brace
90  foreach {endpos startpos} [lreverse $pos(space)] {
91  if {[expr [lsearch [lsort -integer [concat $pos(brackets) $endpos]] $endpos] % 2] == 0} {
92  return [list [string range $str $endpos end] $line.$endpos $line.$endcol $prespace]
93  }
94  }
95 
96  return [list [string range $str $startcol end] $line.$startcol $line.$endcol $prespace]
97 
98  }

101  # Parses the current Emmet snippet found in the current editing buffer.
102  # Returns a three element list containing the generated code, the
103  # starting index of the snippet and the ending index of the snippet.
104  proc expand_abbreviation {tid} {
105 
106  set txt [[ns gui]::current_txt $tid]
107 
108  # Get the language of the current insertion cursor
109  if {[set lang [ctext::get_lang $txt insert]] eq ""} {
110  set lang [[ns syntax]::get_language $txt]
111  }
112 
113  # If the current language is CSS, translate the abbreviation as such
114  if {$lang eq "CSS"} {
115 
116  # Get the abbreviation text, translate it and insert it back into the text
117  if {[regexp {(\S+)$} [$txt get "insert linestart" insert] -> abbr]} {
118  if {![catch { [ns emmet_css]::parse $abbr} str]} {
119  [ns snippets]::insert_snippet_into_current $tid $str "insert-[string length $abbr]c" insert
120  }
121  }
122 
123  } else {
124 
125  # Find the snippet text
126  lassign [get_snippet_text_pos $tid] str startpos endpos prespace
127 
128  # Parse the snippet and if no error, insert the resulting string
129  if {![catch { ::parse_emmet $str $prespace} str]} {
130  [ns snippets]::insert_snippet_into_current $tid $str $startpos $endpos
131  }
132 
133  }
