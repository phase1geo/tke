188 proc ctext::event:xscroll {win clientData args} {
189 
190  variable data
191 
192  if {$clientData == ""} {
193  return
194  }
195 
196  uplevel \#0 $clientData $args
197 
198  lassign $args first last
199 
200  if {$first > 0} {
201  set first_line [lindex [split [$win.t index @0,0] .] 0]
202  set last_line [lindex [split [$win.t index @0,[winfo height $win.t]] .] 0]
203  set longest 0
204  for {set i $first_line} {$i <= $last_line} {incr i} {
205  if {[set len [lindex [split [$win.t index $i.end] .] 1]] > $longest} {
206  set longest $len
207  }
208  }
209  set missing [expr round( ($longest * 7) * $first )]
210  } else {
211  set missing 0
212  }
213 
214  # Adjust the warning width line, if one was requested
215  if {$data($win,config,-warnwidth) ne ""} {
216 
217  # Width is calculated by multiplying the longest line with the length of a single character
218  set newx [expr ($data($win,config,-warnwidth) * 7) - $missing]
219 
220  # Move the vertical bar
221  place $win.t.w -x $newx -relheight 1.0
222 
223  }
