170  ## Displays a widget that allows the user to provide input. This
171  # procedure will block until the user has either provided a response
172  # or has cancelled the input by hitting the escape key.
173  #
174  # \param msg Message to display next to input field (prompt)
175  # \param pvar Reference to variable to store user input to
176  # \param allow_vars If set to 1, variables embedded in string will have
177  # substitutions performed; otherwise, the raw string
178  # will be returned. (Optional)
179  #
180  # \return Returns a list containing two elements. The first element is set to a
181  # 1 if the user provided input; otherwise, returns 0 to indicate that the
182  # user cancelled the input operation. The second item is the user provided
183  # value (if the first value is set to 1).
184  proc get_user_input {interp pname msg pvar {allow_vars 1}} {
185 
186  set var ""
187 
188  if {[gui::get_user_response $msg var -allow_vars $allow_vars]} {
189  $interp eval set $pvar [list $var]
190  return 1
191  }
192 
193  return 0
194

259  ######################################################################
260  ## Adds a buffer to the browser. The first option is the name of the
261  # buffer. The second option is a command to execute once the save
262  # is successful. The remaining arguments are the following options:
263  #
264  #
265  proc add_buffer {interp pname name save_command args} {
266 
267  array set opts [list]
268 
269  # If we have an odd number of arguments, we have an error condition
270  if {[expr [llength $args] % 2] == 1} {
271  return -code error [msgcat::mc "Argument list to api::add_file was not an even key/value pair"]
272  }
273 
274  # Get the options
275  array set opts $args
276 
277  # Change out the gutter commands with interpreter versions
278  if {[info exists opts(-gutters)]} {
279  set new_gutters [list]
280  foreach gutter $opts(-gutters) {
281  set new_sym [list]
282  foreach {symname symopts} [lassign $gutter gutter_name] {
283  set new_symopts [list]
284  foreach {symopt symval} $symopts {
285  switch $symopt {
286  "-onenter" -
287  "-onleave" -
288  "-onclick" {
289  lappend new_symopts $symopt "$interp eval $symval"
290  }
291  default {
292  lappend new_symopts $symopt $symval
293  }
294  }
295  }
296  lappend new_sym $symname $new_symopts
297  }
298  lappend new_gutters [list $gutter_name {*}$new_sym]
299  }
300  set opts(-gutters) $new_gutters
301  }
302 
303  # Set the tags
304  if {[info exists opts(-tags)]} {
305  set tag_list [list]
306  foreach tag $opts(-tags) {
307  lappend tag_list "plugin__${pname}__$tag"
308  }
309  set opts(-tags) $tag_list
310  }
311 
312  # If the save command was specified, add the interpreter evaluation
313  if {$save_command ne ""} {
314  set save_command "$interp eval $save_command"
315  }
316 
317  # Finally, add the buffer
318  gui::add_buffer end $name $save_command {*}[array get opts]
319 
320  # Allow the plugin to manipulate the ctext widget
321  set txt [gui::current_txt]
322  $interp alias $txt $txt
323 
324  return $txt
325 
326  }

328  ######################################################################
329  ## Adds a file to the browser. If the first argument does not start with
330  # a '-' character, the argument is considered to be the name of a file
331  # to add. If no filename is specified, an empty/unnamed file will be added.
332  # All other options are considered to be parameters.
333  #
334  # -savecommand \e command
335  # * Specifies the name of a command to execute after
336  # the file is saved.
337  #
338  # -lock (0|1)
339  # * If set to 0, the file will begin in the unlocked
340  # state (i.e., the user can edit the file immediately).
341  # * If set to 1, the file will begin in the locked state
342  # (i.e., the user must unlock the file to edit it)
343  #
344  # -readonly (0|1)
345  # * If set to 1, the file will be considered readonly
346  # (i.e., the file will be locked indefinitely); otherwise,
347  # the file will be able to be edited.
348  #
349  # -remember (0|1)
350  # * If set to 0, the file will not be saved to the user's session file
351  # when the application is quit. By default, the file will be
352  # remembered and reloaded when the application is reopened.
353  #
354  # -sidebar (0|1)
355  # * If set to 1 (default), the file's directory contents
356  # will be included in the sidebar; otherwise, the file's
357  # directory components will not be added to the sidebar.
358  #
359  # -saveas (0|1)
360  # * If set to 0 (default), the file will be saved to the
361  # current file; otherwise, the file will always force a
362  # save as dialog to be displayed when saving.
363  #
364  # -buffer (0|1)
365  # * If set to 0 (default), the file will be added as a normal file;
366  # however, if set to 1, the file will be treated as a temporary file
367  # that will be automatically deleted when the tab is closed.
368  #
369  # -diff (0|1)
370  # * If set to 0 (default), the file will be added as an editable file;
371  # however, if set to 1, the file will be inserted as a difference viewer,
372  # allowing the user to view file differences visually within the editor.
373  #
374  # -gutters \e list
375  # * Creates a gutter in the editor. The contents of list are as follows:
376  # \code {name {{symbol_name {symbol_tag_options+}}+}}+ \endcode
377  # For a list of valid symbol_tag_options, see the options available for
378  # tags in a text widget.
379  #
380  # -other (0|1)
381  # * If set to 0 (default), the file will be created in a new tab in the
382  # current pane; however, if set to 1, the file will be created in a new
383  # tab in the other pane (the other pane will be created if it does not
384  # exist).
385  #
386  # -tags \e list
387  # * A list of plugin bindtag suffixes that will be applied only to this
388  # this text widget.
389  #
390  # -name \e filename
391  # * If this option is specified when the filename is not specified, it will
392  # add a new tab to the editor whose name matches the given name. If the
393  # user saves the file, the contents will be saved to disk with the given
394  # file name. The given filename does not need to exist prior to calling
395  # this procedure.
396  proc add_file {interp pname args} {
397 
398  set fname ""
399  array set opts [list]
400 
401  # If no filename is given, add a new file to the editor
402  if {([llength $args] > 0) && ([string index [lindex $args 0] 0] ne "-")} {
403 
404  # Peel the filename from the rest of the arguments
405  set args [lassign $args fname]
406 
407  # Check to make sure that the file is safe to add to the editor, and
408  # if it is, create the normalized pathname of the filename.
409  if {[set fname [interpreter::check_file $pname $fname]] eq ""} {
410  return -code error "permission error"
411  }
412 
413  }
414 
415  # If we have an odd number of arguments, we have an error condition
416  if {[expr [llength $args] % 2] == 1} {
417  return -code error [msgcat::mc "Argument list to api::add_file was not an even key/value pair"]
418  }
419 
420  # Get the options
421  array set opts $args
422 
423  # If the -savecommand option was given, wrap it in an interp eval call
424  # so that we don't execute the command in the master interpreter.
425  if {[info exists opts(-savecommand)]} {
426  set opts(-savecommand) "$interp eval $opts(-savecommand)"
427  }
428 
429  # Change out the gutter commands with interpreter versions
430  if {[info exists opts(-gutters)]} {
431  set new_gutters [list]
432  foreach gutter $opts(-gutters) {
433  set new_sym [list]
434  foreach {symname symopts} [lassign $gutter gutter_name] {
435  set new_symopts [list]
436  foreach {symopt symval} $symopts {
437  switch $symopt {
438  "-onenter" -
439  "-onleave" -
440  "-onclick" {
441  lappend new_symopts $symopt "$interp eval $symval"
442  }
443  default {
444  lappend new_symopts $symopt $symval
445  }
446  }
447  }
448  lappend new_sym $symname $new_symopts
449  }
450  lappend new_gutters [list $gutter_name {*}$new_sym]
451  }
452  set opts(-gutters) $new_gutters
453  }
454 
455  # Set the tags
456  if {[info exists opts(-tags)]} {
457  set tag_list [list]
458  foreach tag $opts(-tags) {
459  lappend tag_list "plugin__${pname}__$tag"
460  }
461  set opts(-tags) $tag_list
462  }
463 
464  # Finally, add the new file
465  if {$fname eq ""} {
466  gui::add_new_file end {*}[array get opts]
467  } else {
468  gui::add_file end $fname {*}[array get opts]
469  }
470 
471  # Allow the plugin to manipulate the ctext widget
472  set txt [gui::current_txt]
473  $interp alias $txt $txt
474 
475  return $txt
476 
477  }
478 
479  }

483  ######################################################################
484  ## \return Returns the text widget index based on the given input
485  # parameters.
486  #
487  # \param txt Pathname of text widget to get index of.
488  # \param position The specifies the visible cursor position to lookup. The
489  # values that can be used for this option are as follows:
490  # - left Index num characters left of the starting position, staying on the same line.
491  # - right Index num characters right of the starting position, staying on the same line.
492  # - up Index above the starting position, remaining in the same
493  # column, if possible.
494  # - down Index below the starting position, remaining in the same
495  # column, if possible.
496  # - first Index of the first line/column in the buffer.
497  # - last Index of the last line/column in the buffer.
498  # - char Index of the a specified character before or after the starting
499  # position.
500  # - dchar Index of num'th character before or after the starting
501  # position.
502  # - findchar Index of a specified character before or after the starting
503  # position.
504  # - firstchar Index of first non-whitespace character of the line specified
505  # by startpos.
506  # - lastchar Index of last non-whitespace character of the line specified
507  # by startpos.
508  # - wordstart Index of the first character of the word containing startpos.
509  # - wordend Index of the last character+1 of the word containing startpos.
510  # - WORDstart Index of the first character of the WORD containing startpos.
511  # - WORDend Index of the last character+1 of the WORD containing startpos.
512  # - column Index of the character in the line containing startpos at the
513  # num'th position.
514  # - linenum Index of the first non-whitespace character on the given line.
515  # - linestart Index of the beginning of the line containing startpos.
516  # - lineend Index of the ending of the line containing startpos.
517  # - dispstart Index of the first character that is displayed in the line
518  # containing startpos.
519  # - dispmid Index of the middle-most character that is displayed in the
520  # line containing startpos.
521  # - dispend Index of the last character that is displayed in the line
522  # containing startpos.
523  # - screentop Index of the start of the first line that is displayed in
524  # the buffer.
525  # - screenmid Index of the start of the middle-most line that is displayed
526  # in the buffer.
527  # - screenbot Index of the start of the last line that is displayed in
528  # the buffer.
529  # - numberstart First numerical character of the word containing startpos.
530  # - numberend Last numerical character of the word containing startpos.
531  # - spacestart First whitespace character of the whitespace containing startpos.
532  # - spaceend Last whitespace character of the whitespace containing startpos.
533  # \param args Modifier arguments based on position value.
534  # -dir Specifies direction from starting position (values are "next"
535  # or "prev"). Defaults to "next".
536  # -startpos Specifies the starting index of calculation. Defaults to "insert".
537  # -num Specifies the number to apply. Defaults to 1.
538  # -char Used with "findchar" position type. Specifies the character
539  # to find.
540  # -exclusive If set to 1, returns character position before calculated
541  # index. Defaults to 0.
542  # -column Specifies the name of a variable containing the column to
543  # use for "up" and "down" positions.
544  # -adjust Adjusts the calculated index by the given value before
545  # returning the result.
546  proc get_index {interp pname txt position args} {
547 
548  return [edit::get_index $txt $position {*}$args]
549 
550  }

868  namespace eval preferences {
869 
870  ######################################################################
871  ## Returns a references to a widget created for the preferences window.
872  #
873  # \return Returns the pathname of the widget to pack.
874  #
875  # \param type Specifies the type of widget to create.
876  # (Legal values are: checkbutton, radiobutton, menubutton,
877  # emtry, text, spinbox)
878  # \param win Pathname of parent window to add widgets to.
879  # \param pref Name of preference value associated with the widget.
880  # \param msg Label text to associate with the widget (this text is
881  # searchable.
882  # \param args For all widget types that are not "spacer", the first arg
883  # must be the name of the preference value associated with the
884  # widget, the second arg must be a label to associated with the
885  # widget (this text is searchable), the rest of the arguments
886  # provide additional information required by the widget.
887  proc widget {interp pname type win args} {
888 
889  # Figure out a unique identifier for the widget within the parent frame
890  set index [llength [winfo children $win]]
891 
892  array set opts {
893  -grid 0
894  }
895 
896  switch $type {
897  spacer {
898  array set opts $args
899  return [pref_ui::make_spacer $win $opts(-grid)]
900  }
901  help {
902  if {([llength $args] < 1) || (([llength $args] % 2) == 0)} {
903  return -code error "api::preferences::widget $type sent an incorrect number of parameters"
904  }
905  set args [lassign $args msg]
906  array set opts $args
907  return [pref_ui::make_help $win $msg $opts(-grid)]
908  }
909  default {
910 
911  if {([llength $args] < 2) || (([llength $args] % 2) == 1)} {
912  return -code error "api::preferences::widget $type sent an incorrect number of parameters"
913  }
914 
915  set args [lassign $args pref msg]
916 
917  array set opts {
918  -value ""
919  -values ""
920  -watermark ""
921  -grid 0
922  -from ""
923  -to ""
924  -increment 1
925  -ending ""
926  -color "white"
927  -height 4
928  -columns ""
929  -help ""
930  }
931  array set opts $args
932 
933  # Calculate the full preference pathname
934  set pref_path "Plugins/$pname/$pref"
935 
936  # Make sure that the preference was loaded prior to creating the UI
937  if {![info exists [preferences::ref $pref_path]]} {
938  return -code error "Plugin preference $pref for $pname not previously loaded"
939  }
940 
941  switch $type {
942  checkbutton {
943  return [pref_ui::make_cb $win.cb$index $msg Plugins/$pname/$pref $opts(-grid)]
944  }
945  radiobutton {
946  if {$opts(-value) eq ""} {
947  return -code error "Radiobutton widget must have -value option set"
948  }
949  return [pref_ui::make_rb $win.rb$index $msg Plugins/$pname/$pref $opts(-value) $opts(-grid)]
950  }
951  menubutton {
952  if {$opts(-values) eq ""} {
953  return -code error "Menubutton widget must have -values option set"
954  }
955  return [pref_ui::make_mb $win.mb$index $msg Plugins/$pname/$pref $opts(-values) $opts(-grid)]
956  }
957  entry {
958  return [pref_ui::make_entry $win.e$index $msg Plugins/$pname/$pref $opts(-watermark) $opts(-grid) $opts(-help)]
959  }
960  token {
961  return [pref_ui::make_token $win.te$index $msg Plugins/$pname/$pref $opts(-watermark) $opts(-grid) $opts(-help)]
962  }
963  text {
964  return [pref_ui::make_text $win.t$index $msg Plugins/$pname/$pref $opts(-height) $opts(-grid) $opts(-help)]
965  }
966  spinbox {
967  if {$opts(-from) eq ""} {
968  return -code error "Spinbox widget must have -from option set"
969  }
970  if {$opts(-to) eq ""} {
971  return -code error "Spinbox widget must have -to option set"
972  }
973  return [pref_ui::make_sb $win.sb$index $msg Plugins/$pname/$pref $opts(-from) $opts(-to) $opts(-increment) $opts(-grid) $opts(-ending)]
974  }
975  colorpicker {
976  return [pref_ui::make_cp $win.cp$index $msg Plugins/$pname/$pref $opts(-color) $opts(-grid)]
977  }
978  table {
979  if {$opts(-columns) eq ""} {
980  return -code error "Table widget must have -columns option set"
981  }
982  return [pref_ui::make_table $win.tl$index $msg Plugins/$pname/$pref $opts(-columns) $opts(-height) $opts(-grid) $opts(-help)]
983  }
984  default {
985  return -code error "Unsupported preference widget type ($type)"
986  }
987  }
988 
989  }
990  }
