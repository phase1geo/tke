165  ## Displays a widget that allows the user to provide input. This
166  # procedure will block until the user has either provided a response
167  # or has cancelled the input by hitting the escape key.
168  #
169  # \param msg Message to display next to input field (prompt)
170  # \param pvar Reference to variable to store user input to
171  # \param allow_vars If set to 1, variables embedded in string will have
172  # substitutions performed; otherwise, the raw string
173  # will be returned.
174  #
175  # \return Returns a list containing two elements. The first element is set to a
176  # 1 if the user provided input; otherwise, returns 0 to indicate that the
177  # user cancelled the input operation. The second item is the user provided
178  # value (if the first value is set to 1).
179  proc get_user_input {interp pname msg pvar {allow_vars 1}} {
180 
181  set var ""
182 
183  if {[gui::get_user_response $msg var $allow_vars]} {
184  $interp eval set $pvar [list $var]
185  return 1
186  }
187 
188  return 0
189 
190  }

233  ######################################################################
234  ## Adds a buffer to the browser. The first option is the name of the
235  # buffer. The second option is a command to execute once the save
236  # is successful. The remaining arguments are the following options:
237  #
238  #
239  proc add_buffer {interp pname name save_command args} {
240 
241  array set opts [list]
242 
243  # If we have an odd number of arguments, we have an error condition
244  if {[expr [llength $args] % 2] == 1} {
245  return -code error [msgcat::mc "Argument list to api::add_file was not an even key/value pair"]
246  }
247 
248  # Get the options
249  array set opts $args
250 
251  # Change out the gutter commands with interpreter versions
252  if {[info exists opts(-gutters)]} {
253  set new_gutters [list]
254  foreach gutter $opts(-gutters) {
255  set new_sym [list]
256  foreach {symname symopts} [lassign $gutter gutter_name] {
257  set new_symopts [list]
258  foreach {symopt symval} $symopts {
259  switch $symopt {
260  "-onenter" -
261  "-onleave" -
262  "-onclick" {
263  lappend new_symopts $symopt "$interp eval $symval"
264  }
265  default {
266  lappend new_symopts $symopt $symval
267  }
268  }
269  }
270  lappend new_sym $symname $new_symopts
271  }
272  lappend new_gutters [list $gutter_name {*}$new_sym]
273  }
274  set opts(-gutters) $new_gutters
275  }
276 
277  # Set the tags
278  if {[info exists opts(-tags)]} {
279  set tag_list [list]
280  foreach tag $opts(-tags) {
281  lappend tag_list "plugin__${pname}__$tag"
282  }
283  set opts(-tags) $tag_list
284  }
285 
286  # If the save command was specified, add the interpreter evaluation
287  if {$save_command ne ""} {
288  set save_command "$interp eval $save_command"
289  }
290 
291  # Finally, add the buffer
292  gui::add_buffer end $name $save_command {*}[array get opts]
293 
294  # Allow the plugin to manipulate the ctext widget
295  set txt [gui::current_txt {}]
296  $interp alias $txt $txt
297 
298  return $txt
299 
300  }

302  ######################################################################
303  ## Adds a file to the browser. If the first argument does not start with
304  # a '-' character, the argument is considered to be the name of a file
305  # to add. If no filename is specified, an empty/unnamed file will be added.
306  # All other options are considered to be parameters.
307  #
308  # -savecommand \e command
309  # * Specifies the name of a command to execute after
310  # the file is saved.
311  #
312  # -lock (0|1)
313  # * If set to 0, the file will begin in the unlocked
314  # state (i.e., the user can edit the file immediately).
315  # * If set to 1, the file will begin in the locked state
316  # (i.e., the user must unlock the file to edit it)
317  #
318  # -readonly (0|1)
319  # * If set to 1, the file will be considered readonly
320  # (i.e., the file will be locked indefinitely); otherwise,
321  # the file will be able to be edited.
322  #
323  # -remember (0|1)
324  # * If set to 0, the file will not be saved to the user's session file
325  # when the application is quit. By default, the file will be
326  # remembered and reloaded when the application is reopened.
327  #
328  # -sidebar (0|1)
329  # * If set to 1 (default), the file's directory contents
330  # will be included in the sidebar; otherwise, the file's
331  # directory components will not be added to the sidebar.
332  #
333  # -saveas (0|1)
334  # * If set to 0 (default), the file will be saved to the
335  # current file; otherwise, the file will always force a
336  # save as dialog to be displayed when saving.
337  #
338  # -buffer (0|1)
339  # * If set to 0 (default), the file will be added as a normal file;
340  # however, if set to 1, the file will be treated as a temporary file
341  # that will be automatically deleted when the tab is closed.
342  #
343  # -diff (0|1)
344  # * If set to 0 (default), the file will be added as an editable file;
345  # however, if set to 1, the file will be inserted as a difference viewer,
346  # allowing the user to view file differences visually within the editor.
347  #
348  # -gutters \e list
349  # * Creates a gutter in the editor. The contents of list are as follows:
350  # \code {name {{symbol_name {symbol_tag_options+}}+}}+ \endcode
351  # For a list of valid symbol_tag_options, see the options available for
352  # tags in a text widget.
353  #
354  # -other (0|1)
355  # * If set to 0 (default), the file will be created in a new tab in the
356  # current pane; however, if set to 1, the file will be created in a new
357  # tab in the other pane (the other pane will be created if it does not
358  # exist).
359  #
360  # -tags \e list
361  # * A list of plugin bindtag suffixes that will be applied only to this
362  # this text widget.
363  #
364  # -name \e filename
365  # * If this option is specified when the filename is not specified, it will
366  # add a new tab to the editor whose name matches the given name. If the
367  # user saves the file, the contents will be saved to disk with the given
368  # file name. The given filename does not need to exist prior to calling
369  # this procedure.
370  proc add_file {interp pname args} {
371 
372  set fname ""
373  array set opts [list]
374 
375  # If no filename is given, add a new file to the editor
376  if {([llength $args] > 0) && ([string index [lindex $args 0] 0] ne "-")} {
377 
378  # Peel the filename from the rest of the arguments
379  set args [lassign $args fname]
380 
381  # Check to make sure that the file is safe to add to the editor, and
382  # if it is, create the normalized pathname of the filename.
383  if {[set fname [interpreter::check_file $pname $fname]] eq ""} {
384  return -code error "permission error"
385  }
386 
387  }
388 
389  # If we have an odd number of arguments, we have an error condition
390  if {[expr [llength $args] % 2] == 1} {
391  return -code error [msgcat::mc "Argument list to api::add_file was not an even key/value pair"]
392  }
393 
394  # Get the options
395  array set opts $args
396 
397  # If the -savecommand option was given, wrap it in an interp eval call
398  # so that we don't execute the command in the master interpreter.
399  if {[info exists opts(-savecommand)]} {
400  set opts(-savecommand) "$interp eval $opts(-savecommand)"
401  }
402 
403  # Change out the gutter commands with interpreter versions
404  if {[info exists opts(-gutters)]} {
405  set new_gutters [list]
406  foreach gutter $opts(-gutters) {
407  set new_sym [list]
408  foreach {symname symopts} [lassign $gutter gutter_name] {
409  set new_symopts [list]
410  foreach {symopt symval} $symopts {
411  switch $symopt {
412  "-onenter" -
413  "-onleave" -
414  "-onclick" {
415  lappend new_symopts $symopt "$interp eval $symval"
416  }
417  default {
418  lappend new_symopts $symopt $symval
419  }
420  }
421  }
422  lappend new_sym $symname $new_symopts
423  }
424  lappend new_gutters [list $gutter_name {*}$new_sym]
425  }
426  set opts(-gutters) $new_gutters
427  }
428 
429  # Set the tags
430  if {[info exists opts(-tags)]} {
431  set tag_list [list]
432  foreach tag $opts(-tags) {
433  lappend tag_list "plugin__${pname}__$tag"
434  }
435  set opts(-tags) $tag_list
436  }
437 
438  # Finally, add the new file
439  if {$fname eq ""} {
440  gui::add_new_file end {*}[array get opts]
441  } else {
442  gui::add_file end $fname {*}[array get opts]
443  }
444 
445  # Allow the plugin to manipulate the ctext widget
446  set txt [gui::current_txt {}]
447  $interp alias $txt $txt
448 
449  return $txt
450 
451  }
