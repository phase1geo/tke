44  # Performs a search of the curren text widget in the given direction
45  # with the text specified in the specified entry widget.
46  proc find_start {tid direction} {
47 
48  variable data
49 
50  # Get the current text widget
51  set txt [[ns gui]::current_txt $tid]
52 
53  # Get the search information
54  lassign [set search_data [[ns gui]::get_search_data find]] str case_sensitive saved
55 
56  # If the user has specified a new search value, find all occurrences
57  if {$str ne ""} {
58 
59  # Escape any parenthesis in the regular expression
60  set str [string map {{(} {\(} {)} {\)}} $str]
61 
62  # Test the regular expression, if it is invalid, let the user know
63  if {[catch { regexp $str ""} rc]} {
64  after 100 [list [ns gui]::set_info_message $rc]
65  return
66  }
67 
68  # Gather any search options
69  set search_opts [list]
70  if {!$case_sensitive} {
71  lappend search_opts -nocase
72  }
73 
74  # Save the find text to history
75  add_history find $search_data
76 
77  # Clear the search highlight class
78  find_clear $tid
79 
80  # Create a highlight class for the given search string
81  ctext::addSearchClassForRegexp $txt search black yellow "" $str $search_opts
82 
83  }
84 
85  # Select the search term
86  if {$direction eq "next"} {
87  find_next $txt 0
88  } else {
89  find_prev $txt 0
90  }
91 
92  }

107  # Searches for the next occurrence of the search item.
108  proc find_next {txt app} {
109 
110  set wrapped 0
111 
112  # If we are not appending to the selection, clear the selection
113  if {!$app} {
114  $txt tag remove sel 1.0 end
115  }
116 
117  # Search the text widget from the current insertion cursor forward.
118  lassign [$txt tag nextrange _search "insert+1c"] startpos endpos
119 
120  # We need to wrap on the search item
121  if {$startpos eq ""} {
122  lassign [$txt tag nextrange _search 1.0] startpos endpos
123  set wrapped 1
124  }
125 
126  # Select the next match
127  if {$startpos ne ""} {
128  if {![[ns vim]::in_vim_mode $txt.t]} {
129  $txt tag add sel $startpos $endpos
130  }
131  ::tk::TextSetCursor $txt $startpos
132  if {$wrapped} {
133  [ns gui]::set_info_message [msgcat::mc "Search wrapped to beginning of file"]
134  }
135  } else {
136  [ns gui]::set_info_message [msgcat::mc "No search results found"]
137  }
138 
139  # Closes the search interface
140  [ns gui]::close_search
141 
142  }

228  # Performs a search and replace given the expression,
229  proc replace_do_raw {tid sline eline search replace ignore_case all} {
230 
231  variable lengths
232 
233  # Get the current text widget
234  set txt [[ns gui]::current_txt $tid]
235 
236  # Clear the selection
237  $txt tag remove sel 1.0 end
238 
239  # Create regsub arguments
240  set rs_args [list]
241  if {$ignore_case} {
242  lappend rs_args -nocase
243  }
244 
245  # Get the list of items to replace
246  set indices [$txt search -all -regexp -count [ns search]::lengths {*}$rs_args -- $search $sline $eline]
247 
248  if {$all} {
249  set indices [lreverse $indices]
250  set lengths [lreverse $lengths]
251  } else {
252  set last_line 0
253  set i 0
254  foreach index $indices {
255  set curr_line [lindex [split $index .] 0]
256  if {$curr_line != $last_line} {
257  lappend new_indices $index
258  lappend new_lengths [lindex $lengths $i]
259  set last_line $curr_line
260  }
261  incr i
262  }
263  set indices [lreverse $new_indices]
264  set lengths [lreverse $new_lengths]
265  }
266 
267  # Get the number of indices
268  set num_indices [llength $indices]
269 
270  # Replace the text (perform variable substitutions if necessary)
271  for {set i 0} {$i < $num_indices} {incr i} {
272  set index [lindex $indices $i]
273  $txt replace $index "$index+[lindex $lengths $i]c" [regsub $search [$txt get $index "$index+[lindex $lengths $i]c"] $replace]
274  }
275 
276  if {$num_indices > 0} {
277 
278  # Set the insertion cursor to the last match and make that line visible
279  ::tk::TextSetCursor $txt [lindex $indices 0]
280 
281  # Make sure that the insertion cursor is valid
282  if {[[ns vim]::in_vim_mode $txt]} {
283  [ns vim]::adjust_insert $txt
284  }
285 
286  # Specify the number of substitutions that we did
287  [ns gui]::set_info_message [format "%d %s" $num_indices [msgcat::mc "substitutions done"]]
288 
289  } else {
290 
291  [ns gui]::set_info_message [msgcat::mc "No search results found"]
292 
293  }
294 
295  }

362  # Inserts the results from the find in files egrep execution into the
363  # newly created buffer.
364  proc fif_insert_results {find_expr num_files err result} {
365 
366  # Get the current text widget
367  set txt [[ns gui]::current_txt {}]
368 
369  # Change the text state to allow text to be inserted
370  $txt configure -state normal
371 
372  # Get the last index of the text widget
373  set last_line [$txt index end]
374 
375  # Insert a starting mark
376  $txt insert -moddata ignore end "----\n"
377 
378  if {!$err || ($num_files == 0)} {
379 
380  # Append the results to the text widget
381  $txt insert -moddata ignore end [fif_format $result]
382 
383  # Modify find_expr so that information in results window will match
384  if {[string index $find_expr 0] eq "^"} {
385  set find_expr [string range $find_expr 1 end]
386  }
387 
388  # Highlight and bind the matches
389  $txt tag configure fif -underline 1 -borderwidth 1 -relief raised -foreground black -background yellow
390  set i 0
391  foreach index [$txt search -regexp -all -count find_counts -- $find_expr $last_line] {
392  $txt tag add fif $index "$index + [lindex $find_counts $i]c"
393  $txt tag bind fif <Enter> [list %W configure -cursor [ttk::cursor link]]
394  $txt tag bind fif <Leave> [list %W configure -cursor [$txt cget -cursor]]
395  $txt tag bind fif <ButtonRelease-1> [list [ns search]::fif_handle_click %W %x %y]
396  incr i
397  }
398 
399  bind $txt <Key-space> { if {[search::fif_handle_space %W]} break }
400 
401  } else {
402 
403  $txt insert -moddata ignore end "No matches were found for \"$find_expr\"\n\n\n"
404 
405  }
406 
407  # Make sure that the beginning of the inserted text is in view
408  $txt see end
409  ::tk::TextSetCursor $txt $last_line
410 
411  # Change the state back to disabled
412  $txt configure -state disabled
413 
414  }

417  # Formats the raw egrep data to make it more readable.
418  proc fif_format {data} {
419 
420  set results ""
421  set file_results [list]
422  set last_linenum ""
423  set first_separator 1
424  array set indices {}
425  array set fnames {}
426  set index 0
427  set matches 0
428 
429  foreach line [split $data \n] {
430  if {[regexp {^(.*?)([:-])(\d+)[:-](.*)$} $line -> fname type linenum content]} {
431  set first_separator 1
432  if {![info exists fnames($fname)]} {
433  set fnames($fname) 1
434  if {[llength $file_results] > 0} {
435  if {[string trim [lindex $file_results end]] eq "..."} {
436  set file_results [lrange $file_results 0 end-1]
437  }
438  append results "[join $file_results \n]\n\n"
439  set file_results [list]
440  }
441  lappend file_results " [file normalize $fname]:\n"
442  set last_linenum ""
443  array unset indices
444  }
445  if {$type eq ":"} {
446  if {($last_linenum eq "") || ($linenum > $last_linenum)} {
447  lappend file_results [format " %6d: %s" $linenum $content]
448  set indices($linenum) $index
449  set last_linenum $linenum
450  incr index
451  } else {
452  lset file_results $indices($linenum) [string replace [lindex $file_results $indices($linenum)] 11 11 ":"]
453  }
454  incr matches
455  } else {
456  if {($last_linenum eq "") || ($linenum > $last_linenum)} {
457  lappend file_results [format " %6d %s" $linenum $content]
458  set indices($linenum) $index
459  set last_linenum $linenum
460  incr index
461  }
462  }
463  } elseif {[string trim $line] eq "--"} {
464  if {$first_separator} {
465  set first_separator 0
466  } else {
467  lappend file_results " ..."
468  }
469  }
470  }
471 
472  # Append the last files information to the results string
473  append results "[join $file_results \n]\n\n"
474 
475  return "Found $matches [expr {($matches != 1) ? {matches} : {match}}] in [array size fnames] [expr {([array size fnames] != 1) ? {files} : {file}}]\n\n$results"
476 
477  }

480  # Handles a left-click on a matched pattern in the given text widget.
481  # Causes the matching file to be opened and we jump to the matching line.
482  proc fif_handle_selection {W index} {
483 
484  # Get the line number from the beginning of the line
485  regexp {^\s*(\d+)} [$W get "$index linestart" $index] -> linenum
486 
487  # Get the filename of the line that is clicked
488  set findex [$W search -regexp -backwards -count fif_count -- {^\s*(\w+:)?/.*:$} $index]
489  set fname [$W get $findex "$findex+[expr $fif_count - 1]c"]
490 
491  # Add the file to the file viewer (if necessary)
492  [ns gui]::add_file end [string trim $fname]
493 
494  # Jump to the line and set the cursor to the beginning of the line
495  set txt [[ns gui]::current_txt {}]
496  ::tk::TextSetCursor $txt $linenum.0
497 
498  }

509  # Handles a space bar key hit on a matched pattern in the given text
510  # widget.
511  proc fif_handle_space {W} {
512 
513  # Get the current insertion index
514  set insert [$W index insert]
515 
516  # Check to see if the space bar was hit inside of a tag
517  foreach {first last} [$W tag ranges fif] {
518  if {[$W compare $first <= $insert] && [$W compare $insert < $last]} {
519  fif_handle_selection $W [$W index insert]
520  return 1
521  }
522  }
523 
524  return 0

529  # Adds the given string to the find history list.
530  proc add_history {type hist_info} {
531 
532  variable data
533 
534  # Check to see if the search string exists within the history
535  if {[set index [lsearch -exact -index 0 $data($type,hist) [lindex $hist_info 0]]] != -1} {
536  set data($type,hist) [lreplace $data($type,hist) $index $index]
537 
538  # Otherwise, reduce the size of the find history if adding another element will cause it to overflow
539  } else {
540  foreach index [lrange [lreverse [lsearch -index end -all $data($type,hist) 0]] [[ns preferences]::get {Find/MaxHistory}] end] {
541  set data($type,hist) [lreplace $data($type,hist) $index $index]
542  }
543  }
544 
545  # Save the find text to history
546  lappend data($type,hist) $hist_info
547 
548  # Clear the history pointer
549  set data($type,hist_ptr) [llength $data($type,hist)]
550 
551  # Clear the current find text
552  set data($type,current) ""
553 
554  }

565

579  ######################################################################
580  # Moves backwards or forwards through search history, populating the given
581  # entry widget with the history search result. If we are moving forward
582  # in history such that we fall into the present, the entry field will be
583  # set to any text that was entered prior to traversing history.
584  proc traverse_history {type dir} {
585 
586  variable data
587 
588  # Get the length of the find history list
589  set hlen [llength $data($type,hist)]
590 
591  # If the history pointer is -1, save the current text entered
592  if {$data($type,hist_ptr) == $hlen} {
593  if {$dir == 1} {
594  return
595  }
596  set data($type,current) [[ns gui]::get_search_data $type]
597  }
598 
599  # Update the current pointer
600  if {($data($type,hist_ptr) == 0) && ($dir == -1)} {
601  return
602  }
603 
604  incr data($type,hist_ptr) $dir
605 
606  # If the new history pointer is -1, restore the current text value
607  if {$data($type,hist_ptr) == $hlen} {
608  [ns gui]::set_search_data $type $data($type,current)
609  } else {
610  [ns gui]::set_search_data $type [lindex $data($type,hist) $data($type,hist_ptr)]
611  }
612 
613  }
