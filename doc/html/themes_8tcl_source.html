47  # Loads the theme information.
48  proc load {} {
49 
50  variable files
51  variable themes_dir
52 
53  # Update the user's themes directory
54  update_themes_dir
55 
56  # Trace changes to syntax preference values
57  if {[array size files] == 0} {
58  trace variable preferences::prefs(Appearance/Theme) w themes::handle_theme_change
59  trace variable preferences::prefs(Appearance/Colorize) w themes::handle_colorize_change
60  trace variable preferences::prefs(Appearance/HiddenThemes) w themes::handle_hidden_change
61  }
62 
63  # Reset the files/themes arrays and unregister launcher items
64  array unset files
65  launcher::unregister [msgcat::mc "Theme:*"]
66 
67  # Load the tke_dir theme files
68  set tfiles [utils::glob_install [file join $::tke_dir data themes] *.tketheme]
69 
70  # Load the theme files
71  foreach item [glob -nocomplain -directory $themes_dir -types d *] {
72  if {[file exists [file join $item [file tail $item].tketheme]]} {
73  lappend tfiles [file join $item [file tail $item].tketheme]
74  }
75  }
76 
77  # Get the theme information
78  foreach tfile $tfiles {
79  set name [file rootname [file tail $tfile]]
80  set files($name) $tfile
81  }
82 
83  # Create the launcher items (only display the visible themes)
84  foreach name [get_visible_themes] {
85  launcher::register [format "%s: %s" [msgcat::mc "Theme"] $name] [list theme::load_theme $files($name)] "" [list themes::theme_okay]
86  }
87 
88  # Allow the preferences UI to be updated, if it exists

186  ######################################################################
187  # Imports the contents of the given theme file (which must have either
188  # the .tketheme or .tkethemz file extensions). Imports the theme into
189  # the user directory. Returns the name of the installed .tketheme file
190  # if successful; otherwise, returns the empty string.
191  proc import {parent_win fname} {
192 
193  variable files
194  variable themes_dir
195 
196  # Unzip the file contents
197  if {[catch { zipper::unzip $fname $themes_dir} rc]} {
198  if {[catch { exec -ignorestderr unzip -u $fname -d $themes_dir} rc]} {
199  tk_messageBox -parent $parent_win -icon error -type ok -default ok \
200  -message "Unable to unzip theme file" -detail $rc
201  return ""
202  }
203  }
204 
205  # Reload the available themes
206  load
207 
208  # Return the pathname of the installed .tketheme file
209  return $files([file rootname [file tail $fname]])
210 
211  }

214  # Exports the contents of the given theme to the given .tkethemz
215  # directory.
216  proc export {parent_win theme odir creator website license} {
217 
218  # Create the theme directory
219  file mkdir [set theme_dir [file join $odir $theme]]
220 
221  # Populate the theme directory with the given contents
222  if {![theme::export $theme $theme_dir $creator $website $license]} {
223  tk_messageBox -parent $parent_win -icon error -type ok -default ok \
224  -message "Unable to export theme contents"
225  }
226 
227  # Get the current working directory
228  set pwd [pwd]
229 
230  # Set the current working directory to the user themes directory
231  cd $odir
232 
233  # Perform the archive
234  if {[catch { zipper::list2zip $theme [glob -directory $theme -tails *] [file join $theme.tkethemz]} rc]} {
235  if {[catch { exec -ignorestderr zip -r [file join $theme.tkethemz] $theme} rc]} {
236  tk_messageBox -parent $parent_win -icon error -type ok -default ok \
237  -message "Unable to zip theme file"
238  }
239  }
240 
241  # Restore the current working directory
242  cd $pwd
243 
244  # Delete the theme directory and its contents
245  file delete {*}[glob -nocomplain -directory $theme_dir *]
246  file delete -force $theme_dir
247 
248  }

251  # Batch exports all custom themes to a directory on the Desktop.
252  proc export_custom {{parent_win .}} {
253 
254  variable files
255  variable themes_dir
256 
257  # Create the themes directory
258  set output_dir [file join ~ Desktop UpdatedThemes]
259  set current [theme::get_current_theme]
260 
261  # If the output directory exists, delete it
262  if {[file exists $output_dir]} {
263  file delete -force $output_dir
264  }
265 
266  # Make the output directory
267  file mkdir $output_dir
268 
269  # Load each theme and then export it
270  foreach {name theme_file} [array get files] {
271 
272  # Only consider themes from the themes_dir
273  if {[string compare -length [string length $themes_dir] $theme_file $themes_dir] != 0} {
274  continue
275  }
276 
277  # Initialize some variables
278  set license [file join [file dirname $theme_file] LICENSE]
279 
280  # Load the theme
281  theme::read_tketheme $theme_file
282 
283  # Export the theme to the output directory
284  array set attrs [list creator "" website "" date ""]
285  array set attrs [theme::get_attributions]
286 
287  # Export the loaded theme
288  export $parent_win $name $output_dir $attrs(creator) $attrs(website) $license
289 
290  }
291 
292  # Restore the theme namespace with the current theme contents
293  theme::load_theme $files($current)
294 
295  # Tell the user that the export was successful
296  gui::set_info_message [msgcat::mc "Batch custom theme export completed successfully"]
297 
298  }

312
