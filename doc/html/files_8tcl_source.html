56  # Returns a list of information based on the types of data requested
57  # in the parameters for the given file.
58  proc get_info {from from_type args} {
59 
60  variable files
61  variable fields
62 
63  switch $from_type {
64  tab {
65  set index [lsearch -index $fields(tab) $files $from]
66  }
67  fileindex {
68  set index $from
69  }
70  }
71 
72  # Verify that we found a matching file
73  if {$index == -1} {
74  return -code error "files::get_info, Unable to find file with attribute ($from_type) and value ($from)"
75  }
76 
77  set i 0
78  foreach to_type $args {
79  upvar $to_type type$i
80  if {$to_type eq "fileindex"} {
81  set retval [set type$i $index]
82  } elseif {[info exists fields($to_type)]} {
83  set retval [set type$i [lindex $files $index $fields($to_type)]]
84  } else {
85  return -code error "files::get_info, Unsupported to_type ($to_type)"
86  }
87  incr i
88  }
89 
90  return $retval
91 
92  }

229  # Normalizes the given filename and resolves any NFS mount information if
230  # the specified host is not the current host.
231  proc normalize {host fname} {
232 
233  # Perform a normalization of the file
234  set fname [file normalize $fname]
235 
236  # If the host does not match our host, handle the NFS mount normalization
237  if {$host ne [info hostname]} {
238  array set nfs_mounts [preferences::get NFSMounts]
239  if {[info exists nfs_mounts($host)]} {
240  lassign $nfs_mounts($host) mount_dir shortcut
241  set shortcut_len [string length $shortcut]
242  if {[string equal -length $shortcut_len $shortcut $fname]} {
243  set fname [string replace $fname 0 [expr $shortcut_len - 1] $mount_dir]
244  }
245  }
246  }
247 
248  return $fname
249 
250  }

255  proc check_file {index} {
256

262

271  gui::update_file $index
272  }
273  } else {
274  gui::update_file $index
275  }
276  lset files $index $fields(mtime) $file_mtime
277  }
278  } elseif {$mtime ne ""} {
279  set answer [tk_messageBox -parent . -icon question -message [msgcat::mc "Delete tab?"] \
280  -detail $fname -type yesno -default yes]
281  if {$answer eq "yes"} {

283  } else {
284  lset files $index $fields(mtime) ""

323  lset file_info $fields(lock) $opts(-lock)

325  lset file_info $fields(sidebar) $opts(-sidebar)

328  lset file_info $fields(gutters) $opts(-gutters)

332  lset file_info $fields(remember) $opts(-remember)
333  lset file_info $fields(remote) $opts(-remote)
334  lset file_info $fields(xview) $opts(-xview)
335  lset file_info $fields(yview) $opts(-yview)
336  lset file_info $fields(cursor) $opts(-cursor)
337 
338  if {($opts(-remote) eq "") && !$opts(-buffer)} {
339  lset file_info $fields(eol) [get_eol_translation $fname]
340  } else {
341  lset file_info $fields(eol) [get_eol_translation ""]
342  }

345  lappend files $file_info

364  # Returns the contents of the file located at the given tab. Returns
365  # a value of 1 if the file was successfully loaded; otherwise, returns
366  # 0.
367  proc get_file {tab pcontents} {
368 
369  variable files
370  variable fields
371 
372  get_info $tab tab fileindex fname diff remote
373 
374  # Set the loaded indicator
375  lset files $fileindex $fields(loaded) 1
376 
377  upvar $pcontents contents
378 
379  # Get the file contents
380  if {$remote ne ""} {
381  remote::get_file $remote $fname contents modtime
382  lset files $fileindex $fields(mtime) $modtime
383  } elseif {![catch { open $fname r} rc]} {
384  set contents [string range [read $rc] 0 end-1]
385  close $rc
386  lset files $fileindex $fields(mtime) [file mtime $fname]
387  } else {
388  return 0
389  }
390 
391  return 1
392

591  # Duplicates the given filename.
592  proc duplicate_file {fname remote} {
593 
594  # Create the default name of the duplicate file
595  set dup_fname "[file rootname $fname] Copy[file extension $fname]"
596  set num 1
597  if {$remote eq ""} {
598  while {[file exists $dup_fname]} {
599  set dup_fname "[file rootname $fname] Copy [incr num][file extension $fname]"
600  }
601  if {[catch { file copy $fname $dup_fname} rc]} {
602  return -code error $rc
603  }
604  } else {
605  while {[remote::file_exists $remote $dup_fname]} {
606  set dup_fname "[file rootname $fname] Copy [incr num][file extension $fname]"
607  }
608  if {![remote::duplicate_file $remote $fname $dup_fname]} {
609  return -code error ""
610  }
611  }
612 
613  # Allow any plugins to handle the rename
614  plugins::handle_on_duplicate $fname $dup_fname
615 
616  return $dup_fname

643  # Moves the given file/folder to the trash. If there are any issues,
644  # we will throw an exception.
645  proc move_to_trash {fname isdir} {
646 
647  # Allow any plugins to handle the deletion
648  plugins::handle_on_trash $fname
649 
650  # Move the original directory to the trash
651  switch -glob $::tcl_platform(os) {
652 
653  Darwin {
654  set cmd "tell app \"Finder\" to move the POSIX file \"$fname\" to trash"
655  if {[catch { exec -ignorestderr osascript -e $cmd} rc]} {
656  return -code error $rc
657  }
658  close_tabs $fname $isdir
659  return
660  }
661 
662  Linux* {
663  if {![catch { exec -ignorestderr which gvfs-trash 2>@1}]} {
664  if {[catch { exec -ignorestderr gvfs-trash $fname} rc]} {
665  return -code error $rc
666  }
667  close_tabs $fname $isdir
668  return
669  } elseif {![catch { exec -ignorestderr which kioclient 2>@1}]} {
670  if {[catch { exec -ignorestderr kioclient move $fname trash:/} rc]} {
671  return -code error $rc
672  }
673  close_tabs $fname $isdir
674  return
675  } elseif {[file exists [set trash [file join ~ .local share Trash]]]} {
676  if {[info exists ::env(XDG_DATA_HOME)] && ($::env(XDG_DATA_HOME) ne "") && [file exists $::env(XDG_DATA_HOME)]} {
677  set trash $::env(XDG_DATA_HOME)
678  }
679  set trash_path [get_unique_path [file join $trash files] [file tail $fname]]
680  if {![catch { open [file join $trash info [file tail $trash_path].trashinfo] w} rc]} {
681  puts $rc "\[Trash Info\]"
682  puts $rc "Path=$fname"
683  puts $rc "DeletionDate=[clock format [clock seconds] -format {%Y-%m-%dT%T}]"
684  close $rc
685  } else {
686  return -code error $rc
687  }
688  } elseif {[file exists [set trash [file join ~ .Trash]]]} {
689  set trash_path [get_unique_path [file join $trash files] [file tail $fname]]
690  } else {
691  return -code error [msgcat::mc "Unable to determine how to move to trash"]
692  }
693  }
694 
695  *Win* {
696  set binit [file join $::tke_dir Win binit binit.exe]
697  if {[namespace exists ::freewrap] && [zvfs::exists $binit]} {
698  if {[catch { exec -ignorestderr [freewrap::unpack $binit] $fname} rc]} {
699  return -code error $rc
700  }
701  close_tabs $fname $isdir
702  return
703  } elseif {[file exists $binit]} {
704  if {[catch { exec -ignorestderr $binit $fname} rc]} {
705  return -code error $rc
706  }
707  close_tabs $fname $isdir
708  return
709  } elseif {[file exists [file join C: RECYCLER]]} {
710  set trash_path [file join C: RECYCLER]
711  } elseif {[file exists [file join C: {$Recycle.bin}]]} {
712  set trash_path [file join C: {$Recycle.bin}]
713  } else {
714  return -code error [msgcat::mc "Unable to determine how to move to trash"]
715  }
716  }
717 
718  default {
719  return -code error [msgcat::mc "Unable to determine platform"]
720  }
721 
722  }
723 
724  # Finally, move the file/directory to the trash
725  if {[catch { file rename -force $fname $trash_path} rc]} {
726  return -code error $rc
727  }
728 
729  # Close the opened tabs
730  close_tabs $fname $isdir
731 
732  }
