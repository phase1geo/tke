66  proc save {max shortcuts} {
67 
68  variable user_bindings_file
69 
70  if {![catch { open $user_bindings_file w} rc]} {
71 
72  set last_mnu ""
73 
74  foreach shortcut $shortcuts {
75  set mnu_path [lindex $shortcut 0]
76  set mnu [lindex [split $mnu_path /] 0]
77  if {$mnu ne $last_mnu} {
78  if {$last_mnu ne ""} {
79  puts $rc ""
80  }
81  puts $rc "# [string totitle $mnu] menu bindings"
82  set last_mnu $mnu
83  }
84  puts -nonewline $rc "{$mnu_path}[string repeat { } [expr $max - [string length $mnu_path]]] "
85  if {[lindex $shortcut 1] eq ""} {
86  puts $rc "{}"
87  } else {
88  puts $rc [lindex $shortcut 1]
89  }
90  }
91 
92  # Close the file
93  close $rc
94 
95  # Next, load the file
96  load_file 1
97 
98  }

107  # Polls on the bindings file in the tke home directory. Whenever it
108  # changes modification time, re-read the file and store it in the
109  # menu_bindings array
110  proc load_file {skip_base {dummy 0}} {
111 
112  variable base_bindings_file
113  variable user_bindings_file
114  variable menu_bindings
115 
116  # Remove the existing bindings
117  remove_all_bindings
118 
119  # Read in the base bindings file. Copy it to the user bindings file, if one does not exist.
120  if {!$skip_base} {
121  if {[file exists $user_bindings_file]} {
122  if {![catch { tkedat::read $base_bindings_file 0} rc]} {
123  array set menu_bindings $rc
124  array set reversed [lreverse $rc]
125  }
126  } else {
127  file copy -force $base_bindings_file $::tke_home
128  }
129  }
130 
131  # Read in the user bindings file.
132  if {![catch { tkedat::read $user_bindings_file 0} rc]} {
133 
134  # This block of code removes and default menu bindings that are in use by the user.
135  foreach {mnu binding} $rc {
136  if {[info exists reversed($binding)]} {
137  catch { unset menu_bindings($reversed($binding))}
138  }
139  set menu_bindings($mnu) $binding
140  }
141 
142  # Apply the bindings to the UI
143  apply_all_bindings
144 
145  } else {
146 
147  # Remove all menu bindings if we were unable to read the user bindings file (this file should exist)
148  array unset menu_bindings
149 
150  }
151 
152  }

154  ######################################################################
155  # Applies the current bindings from the configuration file.
156  proc apply_all_bindings {} {
157 
158  variable menu_bindings
159  variable bound_menus
160 
161  array unset bound_menus
162 
163  foreach {mnu_path binding} [array get menu_bindings] {
164  if {$binding eq ""} {
165  continue
166  }
167  set menu_list [split $mnu_path /]
168  if {![catch { menus::get_menu [lrange $menu_list 0 end-1]} mnu]} {
169  if {![catch { $mnu index [msgcat::mc [lindex $menu_list end]]} menu_index] && ($menu_index ne "none")} {
170  set value [list "" "" "" "" ""]
171  if {[string range $binding end-1 end] eq "--"} {
172  set binding [string range $binding 0 end-2]
173  lset value 4 "-"
174  }
175  foreach elem [split $binding -] {
176  lset value [lindex [accelerator_mapping $elem] 0] $elem
177  }
178  set binding [join [concat {*}$value] -]
179  set bound_menus($mnu,$menu_index) $binding
180  $mnu entryconfigure $menu_index -accelerator $binding
181  bind all [accelerator_to_sequence $binding] "menus::invoke $mnu $menu_index; break"
182  }
183  }
184  }
185 
186  }

345  # If the sequence detail is the minus key, this will cause problems with the parser so
346  # remove it and append it at the end of the sequence.
347  if {[string range $accelerator end-1 end] eq "--"} {
348  set append_dash 1
349  set accelerator [string range $accelerator 0 end-2]
350  }
351 
352  # Create the sequence
353  foreach value [split $accelerator -] {
354  if {$alt && !$shift && [info exists alt_mapping([string tolower $value])]} {
355  append sequence $alt_mapping([string tolower $value])
356  } elseif {$alt && $shift && [info exists alt_mapping([string toupper $value])]} {
357  append sequence $alt_mapping([string toupper $value])
358  } elseif {$shift && [info exists shift_mapping($value)]} {
359  append sequence $shift_mapping($value)
360  } elseif {[info exists mapping($value)]} {
361  append sequence $mapping($value)
362  } elseif {$value eq "Shift"} {
363  append sequence "Shift-"
364  set shift 1
365  } elseif {$value eq "Alt"} {
366  set alt 1
367  } elseif {[string length $value] == 1} {
368  if {$alt} {
369  append sequence "Mod2-"
370  }
371  append sequence [string tolower $value]
372  } else {
373  append sequence $value
374  }
375  }
376 
377  if {$append_dash} {
378  append sequence "minus"
379  }
380 
381  append sequence ">"
382 
383  return $sequence
384 
385  }

388  # Maps the given value to the displayed.
389  proc accelerator_mapping {value} {
390 
391  array set map {
392  Ctrl,\u2303 0
393  Alt,\u2325 1
394  Shift,\u21e7 2
395  Cmd,\u2318 3
396  Up,\u2191 4
397  Down,\u2193 4
398  Left,\u2190 4
399  Right,\u2192 4
400  }
401 
402  # Special-case the asterisk character
403  if {($value eq "*") || ($value eq "?")} {
404  return [list 4 $value]
405  }
406 
407  if {[set key [array names map $value,*]] ne ""} {
408  return [list $map($key) [lindex [split $key ,] 1]]
409  } elseif {[set key [array names map *,$value]] ne ""} {
410  return [list $map($key) [lindex [split $key ,] 0]]
411  } elseif {[string length $value] == 2} {
412  return [list 4 [string index $value 1]]
413  } else {
414  return [list 4 $value]
415  }
416 
417  }
