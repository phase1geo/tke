95  # Checks to see if the given file is newer than the file within the
96  # editor. If it is newer, prompt the user to update the file.
97  proc check_file {index} {
98 
99  variable files
100  variable files_index
101 
102  # Get the file information
103  lassign [get_info $index fileindex {tab fname mtime modified}] tab fname mtime modified
104 
105  if {$fname ne ""} {
106  if {[file exists $fname]} {
107  file stat $fname stat
108  if {$mtime != $stat(mtime)} {
109  if {$modified} {
110  set answer [tk_messageBox -parent . -icon question -message [msgcat::mc "Reload file?"] \
111  -detail $fname -type yesno -default yes]
112  if {$answer eq "yes"} {
113  update_file $index
114  }
115  } else {
116  update_file $index
117  }
118  lset files $index $files_index(mtime) $stat(mtime)
119  }
120  } elseif {$mtime ne ""} {
121  set answer [tk_messageBox -parent . -icon question -message [msgcat::mc "Delete tab?"] \
122  -detail $fname -type yesno -default yes]
123  if {$answer eq "yes"} {
124  close_tab {} $tab -check 0
125  } else {
126  lset files $index $files_index(mtime) ""
127  }
128  }
129  }
