4 # Replace the standard notebook tab with one that includes a close
5 # button. To use, create a ttk::notebook widget and set the -style

13  # Tk 8.6 has the Visual Styles element engine on windows. If this is
14  # available we use it to get proper windows close buttons.
15  #
16  proc CreateElements {} {
17  if {[lsearch -exact [ttk::style element names] close] == -1} {
18  if {[catch {
19  # WINDOW WP_SMALLCLOSEBUTTON (19)
20  # WINDOW WP_MDICLOSEBUTTON (20)
21  # WINDOW WP_MDIRESTOREBUTTON (22)
22  ttk::style element create close vsapi \
23  WINDOW 20 {disabled 4 {active pressed} 3 active 2 {} 1}
24  }]} then {
25  CreateImageElements
26  }
27  }
28  }

30  proc CreateImageElements {} {
31  # Create two image based elements to provide buttons to close the
32  # tabs or to display an image which provides user-defined functionality.
33  namespace eval ::img {}
34  set imgdir [file join $::tke_dir lib images]
35  image create bitmap ::img::blank -file [file join $imgdir blank.bmp]
36  image create bitmap ::img::close -file [file join $imgdir close.bmp] -foreground grey50
37  image create bitmap ::img::closeactive -file [file join $imgdir close.bmp] -foreground grey10
38  if {[lsearch -exact [ttk::style element names] close] == -1} {
39  if {[catch {
40  ttk::style element create close image \
41  [list ::img::blank \
42  {active !pressed !disabled} ::img::close \
43  {active pressed !disabled} ::img::closeactive] \
44  -padding 8 -sticky {}
45  } err]} { puts stderr $err}
46  }
47  }
