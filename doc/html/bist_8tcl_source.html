41  proc refresh {args} {
42

44  variable tests
45

47  if {![winfo exists .bistwin]} {
48  return
49  }
50 
51  # Get the list of selected diagnostics in the table
52  set selected [get_selections]
53 
54  # Load all of the BIST files
55  foreach bfile [glob -directory [file join $::tke_dir tests] *.tcl] {
56  if {[catch { source $bfile} rc]} {
57  puts $::errorInfo
58  }
59  }
60 
61  # Gather the list of tests to run
62  set tests [list]
63  foreach ns [namespace children] {
64  lappend tests {*}[info procs ${ns}::run_test*]
65  }
66 
67  # Organize the test items
68  set i 0
69  foreach test $tests {
70  lassign [string map {{::} { }} $test] dummy category name
71  lappend test_array($category) $name
72  incr i
73  }
74 
75  # Clear the tablelist
76  $data(widgets,tbl) delete 0 end
77 
78  # Add the test items to the tablelist
79  foreach category [lsort -dictionary [array names test_array]] {
80  set node [$data(widgets,tbl) insertchild root end [list 1 $category 0 0 0 ""]]
81  $data(widgets,tbl) rowconfigure $node -background grey
82  $data(widgets,tbl) cellconfigure $node,selected -image $data(images,checked)
83  foreach test [lsort -dictionary $test_array($category)] {
84  set cmd [join [list bist $category $test] ::]
85  set child [$data(widgets,tbl) insertchild $node end [list 1 $test 0 0 0 $cmd]]
86  $data(widgets,tbl) cellconfigure $child,selected -image $data(images,checked)
87  }
88  }

90  # Collapse all tests
91  $data(widgets,tbl) collapseall
92 
93  # Sets the given selections
94  set_selections $selected

99  # Runs the built-in self test.
100  proc run {} {
101 
102  variable tests
103  variable data
104  variable run_tests
105 
106  # Specify that the regression should run
107  set data(run) 1
108  set data(done) 0
109 
110  # Initialize a few things first
111  initialize
112 
113  # Get the number of tests available to run
114  set testslen [llength $run_tests]
115  set err 0
116  set pass 0
117  set fail 0
118 
119  # Make sure that the results tab is displayed.
120  $data(widgets,nb) select 2
121 
122  # Allow the BIST to dump output to the output text widget
123  $data(widgets,output) configure -state normal
124  $data(widgets,output) delete 1.0 end
125  $data(widgets,output) configure -state disabled
126 
127  # Initialize the pass and fail widgets
128  $data(widgets,pass) configure -text 0
129  $data(widgets,fail) configure -text 0
130 
131  # Configure UI components
132  $data(widgets,refresh) configure -state disabled
133  $data(widgets,run) configure -state disabled
134  $data(widgets,cancel) configure -state normal
135 
136  update idletasks
137 
138  output "---------------------------------------------\n"
139  output "RUNNING BIST - [clock format [clock seconds]]\n\n"
140 
141  set start_time [clock milliseconds]
142 
143  if {$data(run_mode) eq "iter"} {
144  $data(widgets,total) configure -text [$data(widgets,iters) get]
145  set index 0
146  for {set i 0} {$i < [$data(widgets,iters) get]} {incr i} {
147  output "Iteration [format {%4d} [expr $i + 1]]: "
148  switch $data(iter_mode) {
149  random {
150  if {![run_test [expr int( rand() * $testslen )] pass fail err]} {
151  break
152  }
153  }
154  increment {
155  if {![run_test $index pass fail err]} {
156  break
157  }
158  set index [expr ($index + 1) % $testslen]
159  }
160  decrement {
161  set index [expr ($index == 0) ? ($testslen - 1) : ($index - 1)]
162  if {![run_test $index pass fail err]} {
163  break
164  }
165  }
166  }
167  }
168  } elseif {$data(run_mode) eq "loop"} {
169  $data(widgets,total) configure -text [expr [$data(widgets,loops) get] * $testslen]
170  for {set i 0} {$i < [$data(widgets,loops) get]} {incr i} {
171  set tests [list]
172  for {set j 0} {$j < $testslen} {incr j} {
173  lappend tests $j
174  }
175  switch $data(loop_mode) {
176  random {
177  for {set j 0} {$j < $testslen} {incr j} {
178  set rn [expr int( rand() * $testslen )]
179  set val [lindex $tests $rn]
180  lset tests $rn [lindex $tests $j]
181  lset tests $j $val
182  }
183  }
184  decrement {
185  set tests [lreverse $tests]
186  }
187  }
188  output "\nLoop [expr $i + 1]\n\n"
189  for {set j 0} {$j < $testslen} {incr j} {
190  output "Test [format {%4d} [expr $j + 1]]: "
191  if {![run_test [lindex $tests $j] pass fail err]} {
192  break
193  }
194  }
195  if {!$data(run)} {
196  break
197  }
198  }
199  }
200 
201  set stop_time [clock milliseconds]
202 
203  output "\nPASSED: $pass, FAILED: $fail\n\n"
204  output "Runtime: [runtime_string [expr $stop_time - $start_time]]\n"
205  output "---------------------------------------------"
206 
207  # Configure UI components
208  $data(widgets,refresh) configure -state normal
209  $data(widgets,run) configure -state normal
210  $data(widgets,cancel) configure -state disabled
211 
212  # Wrap things up
213  finish
214 
215  }

217  ######################################################################
218  # Run the given test in the run_tests array.
219  proc run_test {index ppass pfail perr} {
220 
221  upvar $ppass pass
222  upvar $pfail fail
223  upvar $perr err
224 
225  variable data
226  variable run_tests
227 
228  # Get the row and text to run
229  lassign [lindex $run_tests $index] test row
230 
231  # Get the row's parent
232  set par [$data(widgets,tbl) parentkey $row]
233 
234  # Increment the count cell for both the child and parent
235  $data(widgets,tbl) cellconfigure $row,count -text [expr [$data(widgets,tbl) cellcget $row,count -text] + 1]
236  $data(widgets,tbl) cellconfigure $par,count -text [expr [$data(widgets,tbl) cellcget $par,count -text] + 1]
237 
238  output "Running [format {%-40s} $test...] "
239 
240  # Run the diagnostic and track the pass/fail status in the table
241  if {[catch { $test} rc]} {
242  incr fail
243  output " FAILED ($rc)\n"
244  logger::log $::errorInfo
245  $data(widgets,fail) configure -text $fail
246  $data(widgets,tbl) cellconfigure $row,fail -text [expr [$data(widgets,tbl) cellcget $row,fail -text] + 1]
247  $data(widgets,tbl) cellconfigure $par,fail -text [expr [$data(widgets,tbl) cellcget $par,fail -text] + 1]
248  } else {
249  incr pass
250  output " PASSED\n"
251  $data(widgets,pass) configure -text $pass
252  $data(widgets,tbl) cellconfigure $row,pass -text [expr [$data(widgets,tbl) cellcget $row,pass -text] + 1]
253  $data(widgets,tbl) cellconfigure $par,pass -text [expr [$data(widgets,tbl) cellcget $par,pass -text] + 1]
254  }
255 
256  # Allow any user events to be handled
257  update
258 
259  # Specify if we should continue to run
260  return $data(run)
261 
262  }

265  # Returns the runtime string.
266  proc runtime_string {ms} {
267 
268  set hours [expr $ms / 3600000]
269  set minutes [expr ($ms - ($hours * 3600000)) / 60000]
270  set seconds [expr ($ms - ($hours * 3600000) - ($minutes * 60000)) / 1000.0]
271 
272  return "$hours hours, $minutes minutes, $seconds seconds"
273 
274  }
