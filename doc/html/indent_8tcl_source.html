99  # Checks the given text prior to the insertion marker to see if it
100  # matches the unindent expressions. Increment/decrement
101  # accordingly.
102  proc check_indent {txt index} {
103 
104  variable indent_exprs
105 
106  # If the auto-indent feature was disabled, we are in vim start mode, or
107  # the current language doesn't have an indent expression, quit now
108  if {($indent_exprs($txt,mode) ne "IND+") || [[ns vim]::in_vim_mode $txt]} {
109  return
110  }
111 
112  # If the current line contains an unindent expression, is not within a comment or string,
113  # and is preceded in the line by only whitespace, replace the whitespace with the proper
114  # indentation whitespace.
115  if {([set uindex [$txt search -regexp -- "[join $indent_exprs($txt,unindent) |]\$" "$index linestart" $index]] ne "") && \ ![
116  ![ctext::inCommentString $txt $uindex]} {
117  set line [$txt get "$index linestart" $uindex]
118  if {($line ne "") && ([string trim $line] eq "")} {
119  $txt replace "$index linestart" $uindex [get_indent_space $txt 1.0 $index]
120  }
121  }
122 
123  }

126  # Handles a newline character. Returns the character position of the
127  # first line of non-space text.
128  proc newline {txt index} {
129 
130  variable indent_exprs
131 
132  # If the auto-indent feature was disabled, we are in vim start mode,
133  # or the current language doesn't have an indent expression, quit now
134  if {($indent_exprs($txt,mode) eq "OFF") || [[ns vim]::in_vim_mode $txt]} {
135  return
136  }
137 
138  # Get the previous space
139  set prev_space [get_previous_indent_space $txt $index]
140 
141  # If we do not need smart indentation, use the previous space
142  if {$indent_exprs($txt,mode) eq "IND"} {
143 
144  set indent_space $prev_space
145 
146  # Otherwise, do smart indentation
147  } else {
148 
149  # Get the current indentation level
150  set indent_space [get_indent_space $txt 1.0 $index]
151 
152  # Check to see if the previous space is greater than the indent space (if so use it instead)
153  if {[string length $prev_space] > [string length $indent_space]} {
154  set indent_space $prev_space
155  }
156 
157  }
158 
159  # Get the current line
160  set line [$txt get $index "$index lineend"]
161 
162  # Create an index to restore the insertion cursor, if necessary
163  set restore_insert ""
164 
165  # Remove any leading whitespace and update indentation level
166  # (if the first non-whitespace char is a closing bracket)
167  if {[regexp {^( *)(.*)} $line -> whitespace rest] && ($rest ne "")} {
168 
169  # If the first non-whitespace characters match an unindent pattern,
170  # lessen the indentation by one
171  if {[regexp [subst {^[join $indent_exprs($txt,unindent) |]}] $rest]} {
172  $txt insert insert "$indent_space\n"
173  set restore_insert [$txt index insert-1c]
174  if {$indent_exprs($txt,mode) eq "IND+"} {
175  set indent_space [string range $indent_space [[ns preferences]::get Editor/IndentSpaces] end]
176  }
177  }
178 
179  # See if we are deleting a multicursor
180  set mcursor [lsearch [$txt tag names $index] "mcursor"]
181 
182  # Delete the whitespace
183  $txt delete $index "$index+[string length $whitespace]c"
184 
185  # If the newline was from a multicursor, we need to re-add the tag since we have deleted it
186  if {$mcursor != -1} {
187  $txt tag add mcursor $index
188  }
189 
190  }
191 
192  # Insert leading whitespace to match current indentation level
193  if {$indent_space ne ""} {
194  $txt insert $index $indent_space
195  }
196 
197  # If we need to restore the insertion cursor, do it now
198  if {$restore_insert ne ""} {
199  $txt mark set insert $restore_insert
200  }
201 
202  }

205  # Returns the indentation (in number of spaces) of the previous line
206  # of text.
207  proc get_previous_indent_space {txt index} {
208 
209  variable indent_exprs
210 
211  if {($indent_exprs($txt,mode) eq "OFF") || \ [
212  [[ns vim]::in_vim_mode $txt] || \ ([
213  ([lindex [split $index .] 0] == 1)} {
214  return 0
215  }
216 
217  set line_pos [expr [lindex [split [$txt index $index] .] 0] - 1]
218 
219  while {($line_pos > 0) && ([string trim [set line [$txt get "$line_pos.0" "$line_pos.end"]]] eq "")} {
220  incr line_pos -1
221  }
222 
223  if {($line_pos > 0) && [regexp {^( *)(.*)} $line -> whitespace rest]} {
224  return $whitespace
225  } else {
226  return ""
227  }
228 
229  }

232  # This procedure counts the number of tags in the given range.
233  proc get_tag_count {txt tag start end} {
234 
235  variable indent_exprs
236 
237  # Initialize the indent_level
238  set count 0
239 
240  # Count all tags that are not within comments or are escaped
241  while {[set range [$txt tag nextrange _$tag $start $end]] ne ""} {
242  lassign $range index start
243  if {![ctext::inCommentString $txt $index]} {
244  incr count [expr [regexp -all $indent_exprs($txt,$tag) [$txt get $index $start]] - [ctext::isEscaped $txt $index]]
245  }
246  }
247 
248  return $count
249 
250  }

254  # index.
255  proc get_indent_space {txt start end} {
256 
257  # Get the current indentation level
258  set indent_level [expr [get_tag_count $txt indent $start $end] - [get_tag_count $txt unindent $start $end]]
259 
260  return [string repeat " " [expr $indent_level * [[ns preferences]::get Editor/IndentSpaces]]]

265  # Formats the given str based on the indentation information of the text
266  # widget at the current insertion cursor.
267  proc format_text {txt startpos endpos} {
268 
269  variable indent_exprs
270 
271  # If the current language doesn't have indentation enabled, quit now
272  if {$indent_exprs($txt,mode) eq "OFF"} {
273  return
274  }
275 
276  # Get the current position and recalculate the endpos
277  set currpos [$txt index "$startpos linestart"]
278  set endpos [$txt index $endpos]
279 
280  # Update the indentation level at the start of the first text line
281  if {[$txt compare $startpos == 1.0]} {
282  set indent_space ""
283  } else {
284  set indent_space [get_indent_space $txt 1.0 "$startpos-1l lineend"]
285  }
286 
287  # Create the regular expression containing the indent and unindent words
288  set uni_re [join $indent_exprs($txt,unindent) |]
289 
290  # Find the last open brace starting from the current insertion point
291  while {[$txt compare $currpos < $endpos]} {
292 
293  # Get the current line
294  set line [$txt get $currpos "$currpos lineend"]
295 
296  # Remove the leading whitespace and modify it to match the current indentation level
297  if {[regexp {^(\s*)(.*)} $line -> whitespace rest]} {
298  if {[string length $whitespace] > 0} {
299  $txt delete $currpos "$currpos+[string length $whitespace]c"
300  }
301  if {[regexp "^(\\\\)*($uni_re)" $rest -> escapes unindent_match] && \ ![
302  ![expr [string length $escapes] % 2]} {
303  set unindent [[ns preferences]::get Editor/IndentSpaces]
304  } else {
305  set unindent_match ""
306  set unindent 0
307  }
308  if {$indent_space ne ""} {
309  $txt insert $currpos [set indent_space [string range $indent_space $unindent end]]
310  }
311  append indent_space [get_indent_space $txt "$currpos+[expr [string length $unindent_match] + [string length $indent_space]]c" "$currpos lineend"]
312  } else {
313  append indent_space [get_indent_space $txt $currpos "$currpos lineend"]
314  }
315 
316  # Increment the starting position to the next line
317  set currpos [$txt index "$currpos+1l linestart"]
318 
319  }
320 
321  # Perform syntax highlighting
322  [winfo parent $txt] highlight $startpos $endpos
323 
324  }
