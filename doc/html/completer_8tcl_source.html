167  # Returns true if a closing character should be automatically added.
168  # This is called when an opening character is detected.
169  proc add_closing {txtt} {
170 
171  # Get the character at the insertion cursor
172  set ch [$txtt get insert]
173 
174  if {[string is space $ch] || ($ch eq "\}") || ($ch eq "\)") || ($ch eq ">") || ($ch eq "]")} {
175  return 1
176  }
177 
178  return 0
179 
180  }

182  ######################################################################
183  # Returns true if a closing character should be omitted from insertion.
184  # This is called when a closing character is detected.
185  proc skip_closing {txtt type} {
186 
187  return [expr [lsearch [$txtt tag names insert] _${type}R] != -1]

192  # Handles a square bracket.
193  proc add_square {txtt side} {
194 
195  variable complete
196 
197  if {$complete($txtt,[ctext::get_lang $txtt "insert-1c"],square) && ![ctext::inComment $txtt "insert-1c"]} {
198  if {$side eq "right"} {
199  if {[skip_closing $txtt square]} {
200  ::tk::TextSetCursor $txtt "insert+1c"
201  return 1
202  }
203  } else {
204  set ins [$txtt index insert]
205  if {[add_closing $txtt]} {
206  $txtt insert insert "\]"
207  }
208  ::tk::TextSetCursor $txtt $ins
209  }
210  }
211 
212  return 0
213 
214  }

378

380  set lang [ctext::get_lang $txtt insert]
381  switch [$txtt get insert-1c insert+1c] {

383  if {$complete($txtt,$lang,square)} {
384  $txtt delete insert

387  }

389  if {$complete($txtt,$lang,curly)} {
390  $txtt delete insert
391  return
392  }
393  }
394  "<>" {
395  if {$complete($txtt,$lang,angled)} {
396  $txtt delete insert
397  return
398  }
399  }
400  "()" {
401  if {$complete($txtt,$lang,paren)} {
402  $txtt delete insert
403  return
404  }
405  }
406  "\"\"" {
407  if {$complete($txtt,$lang,double)} {
408  $txtt delete insert
409  return
410  }
411  }
412  "''" {
413  if {$complete($txtt,$lang,single)} {
414  $txtt delete insert
415  return
416  }
417  }
418  "``" {
419  if {$complete($txtt,$lang,btick)} {
420  $txtt delete insert
421  return
422  }
423  }
424  }
425  switch [$txtt get insert-1c] {
426  "\[" -
427  "\]" { set delete_check square}
428  "\{" -
429  "\}" { set delete_check curly}
430  "<" -
431  ">" { set delete_check angled}
432  "(" -

440  # Checks all of the matches.
441  proc check_all_brackets {txtt args} {
442 
443  array set opts {
444  -string ""
445  -force 0
446  }
447  array set opts $args
448 
449  # If a string was supplied, only perform bracket check for brackets found in string
450  if {$opts(-string) ne ""} {
451  if {[string map {\{ {} \} {}} $opts(-string)] ne $opts(-string)} { check_brackets $txtt curly $opts(-force)}
452  if {[string map {\[ {} \] {}} $opts(-string)] ne $opts(-string)} { check_brackets $txtt square $opts(-force)}
453  if {[string map {( {} ) {}} $opts(-string)] ne $opts(-string)} { check_brackets $txtt paren $opts(-force)}
454  if {[string map {< {} > {}} $opts(-string)] ne $opts(-string)} { check_brackets $txtt angled $opts(-force)}
455 
456  # Otherwise, check all of the brackets
457  } else {
458  foreach type [list square curly paren angled] {
459  check_brackets $txtt $type $opts(-force)
460  }
461  }
462 
463  }

490  ######################################################################
491  # Checks all matches in the editing buffer.
492  proc check_brackets {txtt stype force} {
493 
494  # Clear missing
495  $txtt tag remove missing:$stype 1.0 end
496 
497  # If the mismcatching char option is cleared, don't continue
498  if {!$force && ![[ns preferences]::get Editor/HighlightMismatchingChar]} {
499  return
500  }
501 
502  set count 0
503  set other ${stype}R
504  set olist [lassign [$txtt tag ranges _$other] ofirst olast]
505  set missing [list]
506 
507  # Perform count for all code containing left stypes
508  foreach {sfirst slast} [$txtt tag ranges _${stype}L] {
509  while {($ofirst ne "") && [$txtt compare $sfirst > $ofirst]} {
510  if {[incr count -[$txtt count -chars $ofirst $olast]] < 0} {
511  lappend missing "$olast+${count}c" $olast
512  set count 0
513  }
514  set olist [lassign $olist ofirst olast]
515  }
516  if {$count == 0} {
517  set start $sfirst
518  }
519  incr count [$txtt count -chars $sfirst $slast]
520  }
521 
522  # Perform count for all right types after the above code
523  while {$ofirst ne ""} {
524  if {[incr count -[$txtt count -chars $ofirst $olast]] < 0} {
525  lappend missing "$olast+${count}c" $olast
526  set count 0
527  }
528  set olist [lassign $olist ofirst olast]
529  }
530 
531  # Highlight all brackets that are missing right stypes
532  while {$count > 0} {
533  lappend missing $start "$start+1c"
534  set start [ctext::get_next_bracket $txtt ${stype}L $start]
535  incr count -1
536  }
537 
538  # Highlight all brackets that are missing left stypes
539  catch { $txtt tag add missing:$stype {*}$missing}
540 
541  }

543  ######################################################################
544  # Places the cursor on the next or previous mismatching bracket and
545  # makes it visible in the editing window. If the -check option is
546  # set, returns 0 to indicate that the given option is invalid; otherwise,
547  # returns 1.
548  proc goto_mismatch {dir args} {
549 
550  array set opts {
551  -check 0
552  }
553  array set opts $args
554 
555  # Get the current text widget
556  set txtt [[ns gui]::current_txt {}].t
557 
558  # If the current text buffer was not highlighted, do it now
559  if {[[ns preferences]::get Editor/HighlightMismatchingChar]} {
560 
561  # Find the previous/next index
562  if {$dir eq "next"} {
563  set index end
564  foreach type [list square curly paren angled] {
565  lassign [$txtt tag nextrange missing:$type "insert+1c"] first
566  if {($first ne "") && [$txtt compare $first < $index]} {
567  set index $first
568  }
569  }
570  } else {
571  set index 1.0
572  foreach type [list square curly paren angled] {
573  lassign [$txtt tag prevrange missing:$type insert] first
574  if {($first ne "") && [$txtt compare $first > $index]} {
575  set index $first
576  }
577  }
578  }
579 
580  # Make sure that the current bracket is in view
581  if {[lsearch [$txtt tag names $index] missing:*] != -1} {
582  if {!$opts(-check)} {
583  ::tk::TextSetCursor $txtt $index
584  $txtt see $index
585  }
586  return 1
587  }
588 
589  }
590 
591  return 0
592
