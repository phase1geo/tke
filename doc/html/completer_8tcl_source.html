128  bind postcomp$txt <Key-braceleft> [list [ns completer]::check_brackets %W curly 0]

163  # Returns true if a closing character should be automatically added.
164  # This is called when an opening character is detected.
165  proc add_closing {txtt} {
166 
167  # Get the character at the insertion cursor
168  set ch [$txtt get insert]
169 
170  if {[string is space $ch] || ($ch eq "\}") || ($ch eq "\)") || ($ch eq ">") || ($ch eq "]")} {
171  return 1
172  }
173 
174  return 0
175 
176  }

179  # Returns true if a closing character should be omitted from insertion.
180  # This is called when a closing character is detected.
181  proc skip_closing {txtt type} {
182 
183  return [expr [lsearch [$txtt tag names insert] _${type}R] != -1]
184

188  # Handles a square bracket.
189  proc add_square {txtt side} {
190 
191  variable complete
192 
193  if {$complete($txtt,[ctext::get_lang $txtt "insert-1c"],square) && ![ctext::inComment $txtt "insert-1c"]} {
194  if {$side eq "right"} {
195  if {[skip_closing $txtt square]} {
196  ::tk::TextSetCursor $txtt "insert+1c"
197  return 1
198  }
199  } else {
200  set ins [$txtt index insert]
201  if {[add_closing $txtt]} {
202  $txtt insert insert "\]"
203  }
204  ::tk::TextSetCursor $txtt $ins
205  }
206  }
207 
208  return 0
209 
210  }

389  }

391  if {$complete($txtt,$lang,angled)} {
392  $txtt delete insert
393  return
394  }
395  }

397  if {$complete($txtt,$lang,paren)} {
398  $txtt delete insert
399  return
400  }
401  }
402  "\"\"" {
403  if {$complete($txtt,$lang,double)} {
404  $txtt delete insert
405  return
406  }
407  }
408  "''" {
409  if {$complete($txtt,$lang,single)} {
410  $txtt delete insert
411  return
412  }
413  }
414  "``" {
415  if {$complete($txtt,$lang,btick)} {
416  $txtt delete insert
417  return
418  }
419  }
420  }
421  switch [$txtt get insert-1c] {
422  "\[" -
423  "\]" { set delete_check square}
424  "\{" -
425  "\}" { set delete_check curly}
426  "<" -
427  ">" { set delete_check angled}

436  # Checks all of the matches.
437  proc check_all_brackets {txtt args} {
438 
439  array set opts {
440  -string ""
441  -force 0
442  }
443  array set opts $args
444 
445  # If a string was supplied, only perform bracket check for brackets found in string
446  if {$opts(-string) ne ""} {
447  if {[string map {\{ {} \} {}} $opts(-string)] ne $opts(-string)} { check_brackets $txtt curly $opts(-force)}
448  if {[string map {\[ {} \] {}} $opts(-string)] ne $opts(-string)} { check_brackets $txtt square $opts(-force)}
449  if {[string map {( {} ) {}} $opts(-string)] ne $opts(-string)} { check_brackets $txtt paren $opts(-force)}
450  if {[string map {< {} > {}} $opts(-string)] ne $opts(-string)} { check_brackets $txtt angled $opts(-force)}
451 
452  # Otherwise, check all of the brackets
453  } else {
454  foreach type [list square curly paren angled] {
455  check_brackets $txtt $type $opts(-force)
456  }
457  }
458 
459  }

474  ######################################################################
475  # Checks all matches in the editing buffer.
476  proc check_brackets {txtt stype force} {
477 
478  # Clear missing
479  $txtt tag remove missing:$stype 1.0 end
480 
481  # If the mismcatching char option is cleared, don't continue
482  if {!$force && ![[ns preferences]::get Editor/HighlightMismatchingChar]} {
483  return
484  }
485 
486  set count 0
487  set other ${stype}R
488  set olist [lassign [$txtt tag ranges _$other] ofirst olast]
489  set missing [list]
490 
491  # Perform count for all code containing left stypes
492  foreach {sfirst slast} [$txtt tag ranges _${stype}L] {
493  while {($ofirst ne "") && [$txtt compare $sfirst > $ofirst]} {
494  if {[incr count -[$txtt count -chars $ofirst $olast]] < 0} {
495  lappend missing "$olast+${count}c" $olast
496  set count 0
497  }
498  set olist [lassign $olist ofirst olast]
499  }
500  if {$count == 0} {
501  set start $sfirst
502  }
503  incr count [$txtt count -chars $sfirst $slast]
504  }
505 
506  # Perform count for all right types after the above code
507  while {$ofirst ne ""} {
508  if {[incr count -[$txtt count -chars $ofirst $olast]] < 0} {
509  lappend missing "$olast+${count}c" $olast
510  set count 0
511  }
512  set olist [lassign $olist ofirst olast]
513  }
514 
515  # Highlight all brackets that are missing right stypes
516  while {$count > 0} {
517  lappend missing $start "$start+1c"
518  set start [ctext::get_next_bracket $txtt ${stype}L $start]
519  incr count -1
520  }
521 
522  # Highlight all brackets that are missing left stypes
523  catch { $txtt tag add missing:$stype {*}$missing}
524 
525  }
526 
527  ######################################################################
528  # Places the cursor on the next or previous mismatching bracket and
529  # makes it visible in the editing window. If the -check option is
530  # set, returns 0 to indicate that the given option is invalid; otherwise,
531  # returns 1.
532  proc goto_mismatch {dir args} {
533 
534  array set opts {
535  -check 0
536  }
537  array set opts $args
538 
539  # Get the current text widget
540  set txtt [[ns gui]::current_txt {}].t
541 
542  # If the current text buffer was not highlighted, do it now
543  if {[[ns preferences]::get Editor/HighlightMismatchingChar]} {
544 
545  # Find the previous/next index
546  if {$dir eq "next"} {
547  set index end
548  foreach type [list square curly paren angled] {
549  lassign [$txtt tag nextrange missing:$type "insert+1c"] first
550  if {($first ne "") && [$txtt compare $first < $index]} {
551  set index $first
552  }
553  }
554  } else {
555  set index 1.0
556  foreach type [list square curly paren angled] {
557  lassign [$txtt tag prevrange missing:$type insert] first
558  if {($first ne "") && [$txtt compare $first > $index]} {
559  set index $first
560  }
561  }
562  }
563 
564  # Make sure that the current bracket is in view
565  if {[lsearch [$txtt tag names $index] missing:*] != -1} {
566  if {!$opts(-check)} {
567  ::tk::TextSetCursor $txtt $index
568  $txtt see $index
569  }
570  return 1
571  }
572 
573  }
574 
575  return 0
576
