106  # Center the wizard window on the screen (including dual monitor setups).
107  proc center_on_screen {} {
108 
109  set swidth [winfo screenwidth .wizwin]
110  set sheight [winfo screenheight .wizwin]
111 
112  # If we have a dual monitor setup, center the window on the first window
113  if {[expr ($swidth.0 / $sheight) > 2]} {
114  set swidth [expr $swidth / 2]
115  }
116 
117  # Place the window in the middle of the screen
118  wm geometry .wizwin +[expr ($swidth / 2) - 320]+[expr ($sheight / 2) - 240]

369  # Creates a text widget that automatically wraps.
370  proc make_text {c x y txt {color black}} {
371 
372  variable locs
373 
374  set id [$c create text $x $y -anchor nw -text "" -fill $color]
375 
376  # Create wrapped text
377  set lines [list]
378  set text_width [expr $locs(right_x) - $x]
379 
380  set line ""
381  foreach word $txt {
382  if {([font measure [$c itemcget $id -font] "$line $word"] < $text_width) && \
383  ($word ne "\n")} {
384  append line " $word"
385  } else {
386  lappend lines [string trim $line]
387  set line $word
388  }
389  }
390 
391  if {$line ne ""} {
392  lappend lines [string trim $line]
393  }
394 
395  $c itemconfigure $id -text [join $lines "\n"]
396 
397  return $id
398 
399  }

418  # Creates a radiobutton.
419  proc make_radiobutton {c x y txt var value command} {
420 
421  variable rbs
422 
423  set csize 11
424 
425  # Create the radiobutton
426  set cid1 [$c create oval $x $y [expr $x + $csize] [expr $y + $csize] -outline black -fill white]
427  set cid [$c create oval [expr $x + 2] [expr $y + 2] [expr $x + $csize - 2] [expr $y + $csize - 2] -outline white -fill white]
428  set tid [$c create text [expr $x + $csize + 10] [expr $y - 2] -text $txt -anchor nw]
429 
430  $c bind $cid1 <Button-1> [list set $var $value]
431  $c bind $cid <Button-1> [list set $var $value]
432  $c bind $tid <Button-1> [list set $var $value]
433 
434  set rbs($var,$value) [list $cid $command]
435 
436  # Make the radiobutton look selected
437  if {[set $var] eq $value} {
438  $c itemconfigure $cid -fill black
439  }
440 
441  trace add variable $var write [list startup::handle_rb_var_change $c]
442 
443  return $cid1
444 
445  }
