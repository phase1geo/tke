34  # Handles a selection of the widget in the multicursor mode.
35  proc handle_selection {W} {
36 
37  variable selected
38 
39  # If we are in multimove Vim mode, return immediately
40  if {[vim::in_multimove $W]} {
41  return
42  }
43 
44  set selected 0
45 
46  if {[llength [set sel [$W tag ranges sel]]] > 2} {
47  set selected 1
48  $W tag remove mcursor 1.0 end
49  foreach {start end} $sel {
50  $W tag add mcursor $start
51  }
52  }
53 
54  }

58  proc handle_escape {W} {
59 
60  if {[set first [lindex [$W tag ranges mcursor] 0]] ne ""} {
61 
62  # If we are not in a multimove, delete the mcursors
63  if {![vim::in_multimove $W] && ([vim::get_edit_mode $W] eq "")} {
64  disable $W
65 
66  # Otherwise, position the insertion cursor on the first multicursor position
67  } else {
68  $W cursor set $first
69  }
70 
71  }

126  # Moves all of the cursors using the positional arguments.
127  proc move {txtt posargs} {
128 
129  array set opts {
130  -num 1
131  }
132  array set opts [lassign $posargs motion]
133 
134  # If the motion is not supported, return now
135  if {[motion_unsupported $txtt $motion]} {
136  return
137  }
138 
139  # Get the existing ranges
140  set ranges [$txtt tag ranges mcursor]
141 
142  # Get the list of new ranges
143  set new_ranges [list]
144  foreach {start end} $ranges {
145  set new_start [$txtt index [$txtt index [list {*}$posargs -startpos $start]]]
146  if {[$txtt compare $new_start == "$new_start lineend"] && [$txtt compare $new_start > "$new_start linestart"]} {
147  set new_start [$txtt index $new_start-1c]
148  }
149  lappend new_ranges $start $new_start
150  }
151 
152  # If any cursors are going to "fall off" an edge, don't perform the move
153  switch $motion {
154  left {
155  foreach {start new_start} $new_ranges {
156  if {([lindex [split $start .] 1] - [lindex [split $new_start .] 1]) < $opts(-num)} {
157  adjust_select $txtt
158  return
159  }
160  }
161  }
162  right {
163  foreach {start new_start} $new_ranges {
164  if {([lindex [split $new_start .] 1] - [lindex [split $start .] 1]) < $opts(-num)} {
165  adjust_select $txtt
166  return
167  }
168  }
169  }
170  up {
171  if {([lindex [split [lindex $new_ranges 0] .] 0] - [lindex [split [lindex $new_ranges 1] .] 0]) < $opts(-num)} {
172  adjust_select $txtt
173  return
174  }
175  }
176  down {
177  if {([lindex [split [lindex $new_ranges end] .] 0] - [lindex [split [lindex $new_ranges end-1] .] 0]) < $opts(-num)} {
178  adjust_select $txtt
179  return
180  }
181  }
182  }
183 
184  # Move the cursors
185  $txtt tag remove mcursor 1.0 end
186  foreach {new_start start} [lreverse $new_ranges] {
187  if {[$txtt compare "$new_start linestart" == "$new_start lineend"]} {
188  [winfo parent $txtt]._t insert "$new_start lineend" " " dspace
189  }
190  adjust_set_and_view $txtt $start $new_start
191  }
192 
193  # Adjust the selection
194  adjust_select $txtt
195 
196  }

348  # Copies any multicursors found in the given text block.
349  proc copy {txt start end} {
350 
351  variable copy_cursors
352 
353  # Current index
354  set current $start
355 
356  # Initialize copy cursor information
357  set copy_cursors($txt,offsets) [list]
358  set copy_cursors($txt,value) [clipboard get]
359 
360  # Get the mcursor offsets from start
361  while {[set index [$txt tag nextrange mcursor $current $end]] ne ""} {
362  lappend copy_cursors($txt,offsets) [$txt count -chars $start [lindex $index 0]]
363  set current [$txt index "[lindex $index 0]+1c"]
364  }
365 
366  }

369  # Adds multicursors to the given pasted text.
370  proc paste {txt start} {
371 
372  variable copy_cursors
373 
374  # Only perform the operation if the stored value matches the clipboard contents
375  if {[info exists copy_cursors($txt,value)] && ($copy_cursors($txt,value) eq [clipboard get])} {
376 
377  # Add the mcursors
378  foreach offset $copy_cursors($txt,offsets) {
379  $txt tag add mcursor "$start+${offset}c"
380  }
381 
382  }
383 
384  }
