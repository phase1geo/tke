37  # Adds bindings for multicursor support to the supplied text widget.
38  proc add_bindings {txt} {
39 
40  # Create tag for the multicursor stuff
41  $txt tag configure mcursor -underline 1
42 
43  # Create multicursor bindings
44  bind mcursor$txt <<Selection>> "[ns multicursor]::handle_selection %W"
45  bind mcursor$txt <Mod2-Button-1> "[ns multicursor]::handle_alt_button1 %W %x %y"
46  bind mcursor$txt <Mod2-Button-$::right_click> "[ns multicursor]::handle_alt_button3 %W %x %y"
47  bind mcursor$txt <Shift-Mod2-ButtonPress-1> "[ns multicursor]::handle_shift_alt_buttonpress1 %W %x %y; break"
48  bind mcursor$txt <Shift-Mod2-B1-Motion> "[ns multicursor]::handle_shift_alt_motion %W %x %y; break"
49  bind mcursor$txt <Shift-Mod2-ButtonRelease-1> "[ns multicursor]::handle_shift_alt_buttonrelease1 %W %x %y; break"
50  bind mcursor$txt <Key-Delete> "if {\[[ns multicursor]::handle_delete %W\]} { break }"
51  bind mcursor$txt <Key-BackSpace> "if {\[[ns multicursor]::handle_backspace %W\]} { break }"
52  bind mcursor$txt <Return> "if {\[[ns multicursor]::handle_return %W\]} { break }"
53  bind mcursor$txt <Any-KeyPress> "if {\[[ns multicursor]::handle_keypress %W %A %K\]} { break }"
54  bind mcursor$txt <Escape> "[ns multicursor]::handle_escape %W"
55  bind mcursor$txt <Button-1> "[ns multicursor]::disable %W"
56 
57  # Add the multicursor bindings to the text widget's bindtags
58  bindtags $txt.t [linsert [bindtags $txt.t] 2 mcursor$txt]
59 
60  }

226  # Set a multicursor at the given index.
227  proc add_cursor {txtt index} {
228 
229  variable cursor_anchor
230 
231  if {[$txtt compare "$index lineend" == $index]} {
232  $txtt insert $index " "
233  }
234 
235  if {[llength [set mcursors [lsearch -inline [$txtt tag names $index] mcursor*]]] == 0} {
236  $txtt tag add mcursor $index
237  } else {
238  $txtt tag remove mcursor $index
239  }
240 
241  # Set the cursor anchor to the current index
242  set cursor_anchor $index
243 
244  }
245 
246  ######################################################################
247  # Set multicursors between the anchor and the current line.
248  proc add_cursors {txtt index} {
249 
250  variable cursor_anchor
251 
252  if {$cursor_anchor ne ""} {
253 
254  # Get the anchor line and column
255  lassign [split [set orig_anchor $cursor_anchor] .] row col
256 
257  # Get the current row
258  set curr_row [lindex [split $index .] 0]
259 
260  # Set the cursor
261  if {$row < $curr_row} {
262  for {set i [expr $row + 1]} {$i <= $curr_row} {incr i} {
263  add_cursor $txtt $i.$col
264  }
265  } else {
266  for {set i $curr_row} {$i < $row} {incr i} {
267  add_cursor $txtt $i.$col
268  }
269  }
270 
271  # Re-set the cursor anchor
272  set cursor_anchor $orig_anchor
273 
274  }
275 
276  }

305  ######################################################################
306  # Adjusts the cursors by the given suffix. The valid values for suffix
307  # are:
308  # +1c - Adjusts the cursors one character to the right.
309  # -1c - Adjusts the cursors one character to the left.
310  # +1l - Adjusts the cursors one line down.
311  # -1l - Adjusts the cursors one line up.
312  #
313  # If the insert value is set to 1 and moving the character would cause
314  # the cursor to be lost (beginning/end of line or beginning/end of file),
315  # a line or character will be inserted and the cursor set to that position.
316  # The inserted text will be given the tag name of "insert_tag".
317  proc adjust {txt suffix {insert 0} {insert_tag ""}} {
318 
319  if {[string index $suffix 0] eq "+"} {
320 
321  # If any of the cursors would "fall off the edge", don't modify any of them
322  if {!$insert && ([string index $suffix end] eq "c")} {
323  foreach {start end} [$txt tag ranges mcursor] {
324  if {[$txt compare $start == "$start lineend-1c"]} {
325  return
326  }
327  }
328  }
329 
330  # Move the cursors
331  switch $suffix {
332  "+1c" {
333  foreach {end start} [lreverse [$txt tag ranges mcursor]] {
334  $txt tag remove mcursor $start
335  if {[$txt compare $start == "$start lineend-1c"]} {
336  if {$insert} {
337  $txt insert "$start+1c" " "
338  if {$insert_tag ne ""} {
339  $txt tag add $insert_tag "$start+1 display char"
340  }
341  adjust_set_and_view $txt $start "$start+1 display char"
342  } else {
343  $txt tag add mcursor $start
344  break
345  }
346  } else {
347  adjust_set_and_view $txt $start "$start+1c"
348  }
349  }
350  }
351  "+1l" {
352  foreach {end start} [lreverse [$txt tag ranges mcursor]] {
353  $txt tag remove mcursor $start
354  if {$insert} {
355  $txt insert "$start lineend" "\n "
356  if {$insert_tag ne ""} {
357  $txt tag add $insert_tag "$start+1 display line linestart"
358  }
359  adjust_set_and_view $txt $start "$start+1 display line linestart"
360  } elseif {[$txt compare $start < "end-1l"]} {
361  adjust_set_and_view $txt $start "$start+1 display l"
362  } else {
363  $txt tag add mcursor $start
364  break
365  }
366  }
367  }
368  }
369 
370  } else {
371 
372  # If any of the cursors would "fall off the edge", don't adjust any of them
373  if {!$insert && ([string index $suffix end] eq "c")} {
374  foreach {start end} [$txt tag ranges mcursor] {
375  if {[$txt compare $start == "$start linestart"]} {
376  return
377  }
378  }
379  }
380 
381  # Adjust the cursors
382  switch $suffix {
383  "-1c" {
384  foreach {start end} [$txt tag ranges mcursor] {
385  $txt tag remove mcursor $start
386  if {[$txt compare $start == "$start linestart"]} {
387  if {$insert} {
388  $txt insert $start " "
389  if {$insert_tag ne ""} {
390  $txt tag add $insert_tag $start
391  }
392  $txt tag add mcursor $start
393  } else {
394  $txt tag add mcursor $start
395  break
396  }
397  } else {
398  adjust_set_and_view $txt $start "$start-1c"
399  }
400  }
401  }
402  "-1l" {
403  foreach {start end} [$txt tag ranges mcursor] {
404  $txt tag remove mcursor $start
405  if {$insert} {
406  $txt insert "$start linestart" " \n"
407  if {$insert_tag ne ""} {
408  $txt tag add $insert_tag "$start linestart"
409  }
410  adjust_set_and_view $txt $start "$start linestart"
411  } elseif {[$txt compare $start >= 2.0]} {
412  adjust_set_and_view $txt $start "$start-1 display line"
413  } else {
414  $txt tag add mcursor $start
415  break
416  }
417  }
418  }
419  }
420 
421  }
422 
423  }

425  ######################################################################
426  # Handles the deletion key. The value of suffix defines what text will
427  # be deleted. The following is a listing of valid values for suffix:
428  # - selected = Forces selected text to be deleted (by default this is detected)
429  # - line = Delete the entire line of the current cursor.
430  # - word = Delete the number of words from the current cursor.
431  # - linestart = Delete the line from the start to the current cursor.
432  # - lineend = Delete the line from the current cursor to the end of the line.
433  # - pattern = Delete if the start of the text matches the given pattern.
434  # - -#type = Delete # of types prior to the cursor to the cursor.
435  # - +#type = Delete from the cursor to # of types after the cursor.
436  proc delete {txtt suffix {data ""}} {
437 
438  variable selected
439 
440  set start 1.0
441  set ranges [list]
442  set do_tags [list]
443  set txt [winfo parent $txtt]
444 
445  # Only perform this if multiple cursors
446  if {[enabled $txtt]} {
447 
448  if {$selected || ($suffix eq "selected")} {
449  while {[set range [$txt tag nextrange sel $start]] ne [list]} {
450  lassign $range start end
451  ctext::comments_chars_deleted $txt $start $end do_tags
452  $txt fastdelete -update 0 $start $end
453  lappend ranges $start $end
454  if {([$txtt compare $start == "$start linestart"]) || \
455  ([$txtt compare $start != "$start lineend"])} {
456  add_cursor $txtt $start
457  set start "$start+2c"
458  } else {
459  add_cursor $txtt "$start-1c"
460  set start "$start+1c"
461  }
462  }
463  set selected 0
464 
465  } elseif {$suffix eq "line"} {
466  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
467  set start [$txt index "[lindex $range 0] linestart"]
468  set end [$txt index "[lindex $range 1] lineend"]
469  ctext::comments_chars_deleted $txt $start $end do_tags
470  $txt fastdelete -update 0 $start $end
471  add_cursor $txt.t $start
472  lappend ranges $start $end
473  set start "$start+2c"
474  }
475 
476  } elseif {$suffix eq "word"} {
477  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
478  set start [$txt index "[lindex $range 0] wordstart"]
479  set end [[ns edit]::get_word $txtt next [expr $data - 1] $start]
480  if {[$txt compare $end > "$start lineend"]} {
481  set end [$txt index "$start lineend"]
482  }
483  ctext::comments_chars_deleted $txt $start $end do_tags
484  $txt fastdelete -update 0 $start $end
485  lappend ranges $start $end
486  if {([$txtt compare $start == "$start linestart"]) || \
487  ([$txtt compare $start != "$start lineend"])} {
488  add_cursor $txtt $start
489  set start "$start+2c"
490  } else {
491  add_cursor $txtt "$start-1c"
492  set start "$start+1c"
493  }
494  }
495 
496  } elseif {$suffix eq "linestart"} {
497  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
498  set start [$txt index "[lindex $range 0] linestart"]
499  set end [lindex $range 0]
500  ctext::comments_chars_deleted $txt $start $end do_tags
501  $txt fastdelete -update 0 $start $end
502  lappend ranges $start $end
503  set start "$start+2c"
504  }
505 
506  } elseif {$suffix eq "lineend"} {
507  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
508  set start [lindex $range 0]
509  set end [$txt index "[lindex $range 0] lineend"]
510  ctext::comments_chars_deleted $txt $start $end do_tags
511  $txt fastdelete -update 0 $start $end
512  lappend ranges $start $end
513  if {([$txtt compare $start == "$start linestart"]) || \
514  ([$txtt compare $start != "$start lineend"])} {
515  add_cursor $txtt $start
516  set start "$start+2c"
517  } else {
518  add_cursor $txtt "$start-1c"
519  set start "$start+1c"
520  }
521  }
522 
523  } elseif {$suffix eq "pattern"} {
524  if {[string index $data 0] eq "^"} {
525  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
526  set start [lindex $range 0]
527  if {[regexp $data [$txt get $start "$start lineend"] match]} {
528  set end [$txt index "$start+[string length $match]c"]
529  ctext::comments_chars_deleted $txt $start $end do_tags
530  $txt fastdelete -update 0 $start $end
531  lappend ranges $start $end
532  if {([$txtt compare $start == "$start linestart"]) || \
533  ([$txtt compare $start != "$start lineend"])} {
534  add_cursor $txtt $start
535  set start "$start+2c"
536  } else {
537  add_cursor $txtt "$start-1c"
538  set start "$start+1c"
539  }
540  } else {
541  set start "$start+2c"
542  }
543  }
544  } else {
545  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
546  set start [lindex $range 0]
547  if {[regexp $data [$txt get "$start linestart" $start] match]} {
548  set start [$txt index "[lindex $range 0]-[string length $match]c"]
549  set end [lindex $range 0]
550  ctext::comments_chars_deleted $txt $start $end do_tags
551  $txt fastdelete -update 0 $start $end
552  lappend ranges $start $end
553  }
554  set start "$start+2c"
555  }
556  }
557 
558  } elseif {[string index $suffix 0] eq "-"} {
559  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
560  set start [lindex $range 0]
561  if {[$txt compare "$start$suffix" < "$start linestart"]} {
562  set start [$txt index "[lindex $range 0] linestart"]
563  } else {
564  set start [$txt index "[lindex $range 0]$suffix"]
565  }
566  set end [lindex $range 0]
567  ctext::comments_chars_deleted $txt $start $end do_tags
568  $txt fastdelete -update 0 $start $end
569  lappend ranges $start $end
570  set start "[lindex $range 0]$suffix+2c"
571  }
572 
573  } else {
574  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
575  set start [lindex $range 0]
576  if {[$txt compare "$start$suffix" >= "$start lineend"]} {
577  set end [$txt index "$start lineend"]
578  } else {
579  set end [$txt index "$start$suffix"]
580  }
581  ctext::comments_chars_deleted $txt $start $end do_tags
582  $txt fastdelete -update 0 $start $end
583  lappend ranges $start $end
584  if {([$txtt compare $start == "$start linestart"]) || \
585  ([$txtt compare $start != "$start lineend"])} {
586  add_cursor $txtt $start
587  set start "$start+2c"
588  } else {
589  add_cursor $txtt "$start-1c"
590  set start "$start+1c"
591  }
592  }
593  }
594 
595  $txt highlight -dotags $do_tags {*}$ranges
596 
597  foreach {start end} $ranges {
598  set linestart [$txt._t index "$start linestart"]
599  set lineend [$txt._t index "$end+1c lineend"]
600  ctext::modified $txt 1 [list delete [list $linestart $lineend] [list]]
601  }
602 
603  event generate $txt.t <<CursorChanged>>
604 
605  return 1
606 
607  }
608 
609  return 0
610 
611  }

614  # Handles the insertion of a printable character.
615  proc insert {txtt value {indent_cmd ""}} {
616 
617  variable selected
618 
619  # Insert the value into the text widget for each of the starting positions
620  if {[enabled $txtt]} {
621  set do_tags [list]
622  set txt [winfo parent $txtt]
623  if {$selected} {
624  foreach {end start} [lreverse [$txtt tag ranges mcursor]] {
625  ctext::comments_chars_deleted $txt $start $end do_tags
626  $txtt fastdelete $start $end
627  $txtt tag add mcursor $start
628  }
629  set selected 0
630  }
631  set start 1.0
632  set ranges [list]
633  set valuelen [string length $value]
634  while {[set range [$txtt tag nextrange mcursor $start]] ne [list]} {
635  set start [lindex $range 0]
636  $txtt fastinsert -update 0 $start $value
637  ctext::comments_do_tag $txt $start "$start+${valuelen}c" do_tags
638  set start "$start+[expr $valuelen + 1]c"
639  lappend ranges {*}$range
640  }
641  $txtt highlight -insert 1 -dotags $do_tags {*}$ranges
642  foreach {start end} $ranges {
643  ctext::modified $txt 1 [list insert [list $start $end] [list]]
644  }
645  if {$indent_cmd ne ""} {
646  set start 1.0
647  while {[set range [$txtt tag nextrange mcursor $start]] ne [list]} {
648  set start [$indent_cmd $txtt [lindex $range 0] 0]+2c
649  }
650  } else {
651  event generate $txtt <<CursorChanged>>
652  }
653  return 1
654  }
655 
656  return 0
657 
658  }

661  # Handle the replacement of a given character.
662  proc replace {txtt value {indent_cmd ""}} {
663 
664  variable selected
665 
666  set txt [winfo parent $txtt]
667 
668  # Replace the current insertion cursor with the given value
669  if {[enabled $txt]} {
670  if {$selected} {
671  return [insert $txt $value $indent_cmd]
672  } else {
673  set start 1.0
674  set do_tags [list]
675  set valuelen [string length $value]
676  while {[set range [$txt tag nextrange mcursor $start]] ne [list]} {
677  lassign $range start end
678  ctext::comments_chars_deleted $txt $start $end do_tags
679  $txt fastreplace -update 0 $start "$start+1c" $value
680  ctext::comments_do_tag $txt $start "$start+${valuelen}c" do_tags
681  $txt tag add mcursor "$start+${valuelen}c"
682  set start "$start+[expr $valuelen + 1]c"
683  lappend ranges {*}$range
684  }
685  $txt highlight -insert 1 -dotags $do_tags {*}$ranges
686  foreach {start end} $ranges {
687  ctext::modified $txt 1 [list delete [list $start $end] [list]]
688  ctext::modified $txt 1 [list insert [list $start $end] [list]]
689  }

693  set start [$indent_cmd $txtt [lindex $range 0] 0]+2c
694  }

697  }
698  return 1

706  ######################################################################
707  # Parses the given number string with the format of:
708  # (d|o|x)?<number>+
709  # Where d means to parse and insert decimal numbers, o means to parse
710  # and insert octal numbers, and x means to parse and insert hexidecimal
711  # numbers. If d, o or x are not specified, d is assumed.
712  # Numbers will be inserted at each cursor location such that the first
713  # cursor will be replaced with the number specified by <number>+ and
714  # each successive cursor will have an incrementing value inserted
715  # at its location.
716  proc insert_numbers {txt numstr} {
717 
718  variable selected
719 
720  # If the number string is a decimal number without a preceding 'd' character, add it now
721  if {[set d_added [regexp {^[0-9]+$} $numstr]]} {
722  set numstr "d$numstr"
723  }
724 
725  # Parse the number string to verify that it's valid
726  if {[regexp {^(.*)((b[0-1]*)|(d[0-9]*)|(o[0-7]*)|([xh][0-9a-fA-F]*))$} $numstr -> prefix numstr]} {
727 
728  # Get the cursors
729  set mcursors [lreverse [$txt tag ranges mcursor]]
730 
731  # Get the last number
732  set num_mcursors [expr ([llength $mcursors] / 2)]
733 
734  # If things were selected, delete their characters and re-add the multicursors
735  if {$selected} {
736  foreach {end start} $mcursors {
737  $txt delete $start $end
738  $txt tag add mcursor $start
739  }
740  set selected 0
741  }
742 
743  # Get the number portion of the number string. If one does not exist,
744  # default the number to 0.
745  if {[set num [string range $numstr 1 end]] eq ""} {
746  set num 0
747  }
748 
749  # Handle the value insertions
750  switch [string tolower [string index $numstr 0]] {
751  b {
752  set num [expr 0b$num + ($num_mcursors - 1)]
753  foreach {end start} $mcursors {
754  set binRep [binary format c $num]
755  binary scan $binRep B* binStr
756  $txt insert $start [format "%s%s%s%s" $prefix [string index $numstr 0] [string trimleft [string range $binStr 0 end-1] 0] [string index $binStr end]]
757  incr num -1
758  }
759  }
760  d {
761  set num [expr $num + ($num_mcursors - 1)]
762  foreach {end start} $mcursors {
763  $txt insert $start [format "%s%s%d" $prefix [expr {$d_added ? "" : [string index $numstr 0]}] $num]
764  incr num -1
765  }
766  }
767  o {
768  set num [expr 0o$num + ($num_mcursors - 1)]
769  foreach {end start} $mcursors {
770  $txt insert $start [format "%s%s%o" $prefix [string index $numstr 0] $num]
771  incr num -1
772  }
773  }
774  h -
775  x {
776  set num [expr 0x$num + ($num_mcursors - 1)]
777  foreach {end start} $mcursors {
778  $txt insert $start [format "%s%s%x" $prefix [string index $numstr 0] $num]
779  incr num -1
780  }
781  }
782  }
783 
784  return 1
785 
786  }
787 
788  return 0
789 
790  }

793  # Aligns all of the cursors by inserting spaces prior to each cursor
794  # that is less than the one in the highest column position. If multiple
795  # cursors exist on the same line, the cursor in the lowest column position
796  # is used.
797  proc align {txt} {
798 
799  set last_row -1
800  set max_col 0
801  set cursors [list]
802 
803  # Find the cursor position to align to and the cursors to align
804  foreach {start end} [$txt tag ranges mcursor] {
805  lassign [split $start .] row col
806  if {$row ne $last_row} {
807  set last_row $row
808  if {$col > $max_col} {
809  set max_col $col
810  }
811  lappend cursors [list $row $col]
812  }
813  }
814 
815  # Insert spaces to align all columns
816  foreach cursor $cursors {
817  $txt insert [join $cursor .] [string repeat " " [expr $max_col - [lindex $cursor 1]]]
818  }
819 
820  }
