90  # Adds any found folds to the gutter
91  proc add_folds {txt startpos endpos} {
92 
93  # Get the starting and ending line
94  set startline [lindex [split [$txt index $startpos] .] 0]
95  set endline [lindex [split [$txt index $endpos] .] 0]
96  set lines(open) [list]
97  set lines(end) [list]
98 
99  # Clear the folding gutter in
100  $txt gutter clear folding $startline $endline
101 
102  # Add the folding indicators
103  for {set i $startline} {$i <= $endline} {incr i} {
104  lappend lines([check_fold $txt $i]) $i
105  }
106 
107  $txt gutter set folding open $lines(open) end $lines(end)
108 
109  }

113  proc check_fold {txt line} {
114 
115  set indent_cnt [[ns indent]::get_tag_count $txt.t indent $line.0 $line.end]
116  set unindent_cnt [[ns indent]::get_tag_count $txt.t unindent $line.0 $line.end]
117 
118  return [expr {($indent_cnt > $unindent_cnt) ? "open" : ($indent_cnt < $unindent_cnt) ? "end" : ""}]

120  }

124  proc get_gutter_info {txt} {
125

206  # Closes all open folds.
207  proc close_all_folds {txt} {
208

210

212  set ranges [list]
213  set count 0
214  foreach {tline tag} [concat {*}[get_gutter_info $txt]] {
215  if {($count == 0) && ($tag eq "open")} {
216  set oline [expr $tline + 1]
217  }
218  if {[incr count $inc($tag)] == 0} {
219  lappend ranges $oline.0 $tline.0

223  }
224  }

235  # Opens a fold, showing the contents.
236  proc open_fold {txt line} {
237 
238  set index [expr [lsearch [set closed [$txt gutter get folding close]] $line] + 1]
239 
240  # Get the tag range
241  lassign [$txt tag nextrange _folded $line.0] startpos endpos
242 
243  # Remove the folded tag
244  $txt tag remove _folded $startpos $endpos
245 
246  # Close all of the previous folds
247  foreach tline [lrange $closed $index end] {
248  if {[$txt compare $tline.0 < $endpos]} {
249  close_fold $txt $tline
250  }
251  }
252 
253  # Replace the close symbol with the open symbol
254  $txt gutter clear folding $line
255  $txt gutter set folding open $line
256 
257  }
