46  # Returns the menu associated with the given menu path. The menu path
47  # should be a list where each menu item is the name of the menu within
48  # the path. The final menu in the path should not be included in the
49  # path.
50  proc get_menu {menu_path} {
51 
52  set parent .menubar
53 
54  foreach mnu $menu_path {
55  if {([set mnu_index [$parent index [msgcat::mc $mnu]]] eq "none") || ([$parent type $mnu_index] ne "cascade")} {
56  return -code error "Invalid menu path specified"
57  }
58  set parent [$parent entrycget $mnu_index -menu]
59  }
60 
61  return $parent
62 
63  }

247  # Returns a menu that will be used in the main menu system.
248  proc make_menu {w args} {
249 
250  # Create the menu
251  set mnu [menu $w {*}$args]
252 
253  # Create menu binding that will allow us to Shift click menu items to
254  # edit their shortcuts in the preferences window.
255  bind $mnu <Control-Button-1> { menus::handle_menu_shift_click %W %y; break }
256  bind $mnu <Control-ButtonRelease-1> { break }
257 
258  return $mnu
259 
260  }

408  # Called prior to the file menu posting.
409  proc file_posting {mb} {
410 
411  # Get information for current file
412  if {![catch { gui::get_info {} current fileindex fname readonly lock diff buffer modified remote}]} {
413 
414  # Get state if the file is a buffer
415  set buffer_state [expr {($buffer || $diff) ? "disabled" : "normal"}]
416 
417  # Get the state if the current editing buffer is a difference buffer
418  set diff_state [expr {$diff ? "disabled" : "normal"}]
419 
420  # Get the state for items that are not valid for remote files
421  set no_remote_state [expr {($remote eq "") ? $buffer_state : "disabled"}]
422 
423  # Get the current favorite status
424  set favorite [favorites::is_favorite $fname]
425 
426  # Configure the Lock/Unlock menu item
427  if {$lock} {
428  if {![catch { $mb index [msgcat::mc "Lock"]} index]} {
429  $mb entryconfigure $index -label [msgcat::mc "Unlock"] -state $diff_state -command "menus::unlock_command $mb"
430  }
431  $mb entryconfigure [msgcat::mc "Unlock"] -state [expr {$readonly ? "disabled" : $diff_state}]
432  } else {
433  if {![catch { $mb index [msgcat::mc "Unlock"]} index]} {
434  $mb entryconfigure $index -label [msgcat::mc "Lock"] -state $diff_state -command "menus::lock_command $mb"
435  }
436  $mb entryconfigure [msgcat::mc "Lock"] -state [expr {$readonly ? "disabled" : $diff_state}]
437  }
438 
439  # Configure the Favorite/Unfavorite menu item
440  if {$favorite} {
441  if {![catch { $mb index [msgcat::mc "Favorite"]} index]} {
442  $mb entryconfigure $index -label [msgcat::mc "Unfavorite"] -command "menus::unfavorite_command $mb"
443  }
444  $mb entryconfigure [msgcat::mc "Unfavorite"] -state [expr {(($fname ne "") && !$diff) ? $no_remote_state : "disabled"}]
445  } else {
446  if {![catch { $mb index [msgcat::mc "Unfavorite"]} index]} {
447  $mb entryconfigure $index -label [msgcat::mc "Favorite"] -command "menus::favorite_command $mb"
448  }
449  $mb entryconfigure [msgcat::mc "Favorite"] -state [expr {(($fname ne "") && !$diff) ? $no_remote_state : "disabled"}]
450  }
451 
452  # Configure the Delete/Move To Trash
453  if {($remote eq "") && [preferences::get General/UseMoveToTrash]} {
454  if {![catch { $mb index [msgcat::mc "Delete"]} index]} {
455  $mb entryconfigure $index -label [msgcat::mc "Move To Trash"] -command [list menus::move_to_trash_command]
456  }
457  $mb entryconfigure [msgcat::mc "Move To Trash"] -state $buffer_state
458  } else {
459  if {![catch { $mb index [msgcat::mc "Move To Trash"]} index]} {
460  $mb entryconfigure $index -label [msgcat::mc "Delete"] -command [list menus::delete_command]
461  }
462  $mb entryconfigure [msgcat::mc "Delete"] -state $buffer_state
463  }
464 
465  # Make sure that the file-specific items are enabled
466  $mb entryconfigure [msgcat::mc "Reopen File"] -state $buffer_state
467  $mb entryconfigure [msgcat::mc "Show File Difference"] -state $no_remote_state
468  $mb entryconfigure [msgcat::mc "Save"] -state [expr {$modified ? $diff_state : "disabled"}]
469  $mb entryconfigure [format "%s..." [msgcat::mc "Save As"]] -state $diff_state
470  $mb entryconfigure [format "%s..." [msgcat::mc "Save As Remote"]] -state $diff_state
471  $mb entryconfigure [format "%s..." [msgcat::mc "Save As Template"]] -state $diff_state
472  $mb entryconfigure [format "%s..." [msgcat::mc "Save Selection As"]] -state [expr {[gui::selected] ? "normal" : "disabled"}]
473  $mb entryconfigure [msgcat::mc "Save All"] -state normal
474  $mb entryconfigure [format "%s..." [msgcat::mc "Export"]] -state $diff_state
475  $mb entryconfigure [msgcat::mc "Line Ending"] -state $diff_state
476  $mb entryconfigure [msgcat::mc "Rename"] -state $buffer_state
477  $mb entryconfigure [msgcat::mc "Duplicate"] -state $buffer_state
478  $mb entryconfigure [msgcat::mc "Close"] -state normal
479  $mb entryconfigure [msgcat::mc "Close All"] -state normal
480 
481  } else {
482 
483  # Disable file menu items associated with current tab (since one doesn't currently exist)
484  $mb entryconfigure [msgcat::mc "Reopen File"] -state disabled
485  $mb entryconfigure [msgcat::mc "Show File Difference"] -state disabled
486  $mb entryconfigure [msgcat::mc "Save"] -state disabled
487  $mb entryconfigure [format "%s..." [msgcat::mc "Save As"]] -state disabled
488  $mb entryconfigure [format "%s..." [msgcat::mc "Save As Remote"]] -state disabled
489  $mb entryconfigure [format "%s..." [msgcat::mc "Save As Template"]] -state disabled
