822  proc parse {str} {
823 
824  set lines [list]
825  set line [list]
826 
827  # Set the string to lowercase
828  set str [string tolower $str]
829 
830  while {[string length $str] > 0} {
831  if {[regexp {^-([wmso]+)-([a-z:]+\+?|[a-z]+:-)(.*)$} $str -> prefix word str]} {
832  lappend line word $word prefix $prefix
833  } elseif {[regexp {^-([a-z:]+\+?|[a-z]+:-)(.*)$} $str -> word str]} {
834  lappend line hyphen 1 word $word
835  } elseif {[regexp {^([a-z:]+\+?|[a-z]+:-)(.*)$} $str -> word str]} {
836  lappend line word $word
837  } elseif {[regexp {^(-?[0-9]+\.[0-9]+)(.*)$} $str -> num str]} {
838  lappend line float $num
839  } elseif {[regexp {^(-?[0-9]+)(.*)$} $str -> num str]} {
840  lappend line number $num
841  } elseif {[regexp {^#([0-9a-fA-F]+)(.*)$} $str -> color str]} {
842  switch [string length $color] {
843  1 { lappend line color #[string repeat $color 6]}
844  2 { lappend line color #[string repeat $color 3]}
845  3 { lappend line color #[string repeat [string index $color 0] 2][string repeat [string index $color 1] 2][string repeat [string index $color 2] 2]}
846  6 { lappend line color #$color}
847  }
848  } elseif {[regexp {^!(.*)$} $str -> str]} {
849  lappend line important {}
850  } else {
851  if {$line ne [list]} {
852  lappend lines [generate_line $line]
853  }
854  set line [list]
855  set str [string range $str 1 end]
856  }
857  }
858 
859  if {$line ne ""} {
860  lappend lines [generate_line $line]
861  }
