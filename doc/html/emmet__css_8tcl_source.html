823  # Parses the given Emmet string and returns the generated code version.
824  proc parse {str} {
825 
826  set lines [list]
827  set line [list]
828 
829  # Set the string to lowercase
830  set str [string tolower $str]
831 
832  while {[string length $str] > 0} {
833  if {[regexp {^-([wmso]+)-([a-z:]+\+?|[a-z]+:-)(.*)$} $str -> prefix word str]} {
834  lappend line word $word prefix $prefix
835  } elseif {[regexp {^-([a-z:]+\+?|[a-z]+:-)(.*)$} $str -> word str]} {
836  lappend line hyphen 1 word $word
837  } elseif {[regexp {^([a-z:]+\+?|[a-z]+:-)(.*)$} $str -> word str]} {
838  lappend line word $word
839  } elseif {[regexp {^(-?[0-9]+\.[0-9]+)(.*)$} $str -> num str]} {
840  lappend line float $num
841  } elseif {[regexp {^(-?[0-9]+)(.*)$} $str -> num str]} {
842  lappend line number $num
843  } elseif {[regexp {^#([0-9a-fA-F]+)(.*)$} $str -> color str]} {
844  switch [string length $color] {
845  1 { lappend line color #[string repeat $color 6]}
846  2 { lappend line color #[string repeat $color 3]}
847  3 { lappend line color #[string repeat [string index $color 0] 2][string repeat [string index $color 1] 2][string repeat [string index $color 2] 2]}
848  6 { lappend line color #$color}
849  }
850  } elseif {[regexp {^!(.*)$} $str -> str]} {
851  lappend line important {}
852  } else {
853  if {$line ne [list]} {
854  lappend lines [generate_line $line]
855  }
856  set line [list]
857  set str [string range $str 1 end]
858  }
859  }
860 
861  if {$line ne ""} {
862  lappend lines [generate_line $line]
863  }
864 
865  return [join $lines \n]
866 
867  }

870  # Returns the CSS that is generates from the given line list.
871  proc generate_line {line_list} {
872 
873  variable lookup
874  variable unitless
875  variable prefixes
876  variable keyword_aliases
877 
878  set line [list]
879  set important 0
880  set property ""
881  set values ""
882  set val ""
883  set suffix ""
884  set suffix_needed 0
885  set prefix_list [list]
886  set hyphen 0
887 
888  foreach {type value} $line_list {
889  switch $type {
890  number -
891  float {
892  set val $value
893  if {($property eq "") || ![info exists unitless($property)]} {
894  if {$type eq "number"} {
895  set suffix [[ns preferences]::get {Emmet/CSSIntUnit}]
896  } else {
897  set suffix [[ns preferences]::get {Emmet/CSSFloatUnit}]
898  }
899  set suffix_needed 1
900  } else {
901  set values [insert_value $values $val]
902  }
903  }
904  hyphen {
905  set hyphen 1
906  }
907  prefix {
908  set prefix_list [list]
909  foreach prefix [split $value {}] {
910  if {[info exists prefixes($prefix)]} {
911  lappend prefix_list $prefixes($prefix)
912  }
913  }
914  }
915  word {
916  if {$suffix_needed} {
917  switch $value {
918  - { set suffix "px"}
919  p { set suffix "%"}
920  e { set suffix "em"}
921  x { set suffix "ex"}
922  r { set suffix "rem"}
923  default { set suffix $value}
924  }
925  set suffix_needed 0
926  if {$property ne ""} {
927  set values [insert_value $values $val$suffix]
928  } else {
929  lappend line $value$suffix
930  }
931  } elseif {[info exists lookup($value)]} {
932  lassign $lookup($value) property values
933  set prefix_list [get_prefix_list $property $hyphen]
934  } elseif {[[ns preferences]::get {Emmet/CSSFuzzySearch}] && \
935  [set tmp [lsearch -glob -inline [lsort [array names lookup]] [join [split $value {}] *]]] ne ""} {
936  lassign $lookup($tmp) property values
937  set prefix_list [get_prefix_list $property $hyphen]
938  } elseif {[info exists keyword_aliases($value)]} {
939  set values [insert_value $values $keyword_aliases($value)]
940  } elseif {$property ne ""} {
941  set values [insert_value $values $value]
942  } else {
943  lappend line $value
944  }
945  }
946  color {
947  if {$suffix_needed} {
948  if {$property ne ""} {
949  set values [insert_value $values $val$suffix]
950  } else {
951  lappend line $val$suffix
952  }
953  set suffix_needed 0
954  }
955  switch [string tolower [[ns preferences]::get {Emmet/CSSColorCase}]] {
956  upper { set value [string toupper $value]}
957  lower { set value [string tolower $value]}
958  }
959  if {[[ns preferences]::get {Emmet/CSSColorShort}]} {
960  set value [shorten_color $value]
961  }
962  if {$property ne ""} {
963  set values [insert_value $values $value]
964  } else {
965  lappend line $value
966  }
967  }
968  important {
969  set important 1
970  }
971  }
972  }
973 
974  if {$suffix_needed} {
975  if {$property ne ""} {
976  set values [insert_value $values $val$suffix]
977  } else {
978  lappend line $val$suffix
979  }
980  }
981 
982  if {$important} {
983  if {$property ne ""} {
984  set values [insert_value $values "!important"]
985  } else {
986  lappend line "!important"
987  }
988  }
989 
990  if {$property ne ""} {
991  set index 1
992  while {[regexp {(.*?)\{\|(.*?)\}(.*)$} $values -> before value after]} {
993  if {$value eq ""} {
994  set values "$before\$$index$after"
995  } else {
996  set values "$before\${$index:$value}$after"
997  }
998  incr index
999  }
1000  set lines [list]
1001  set endstr [[ns preferences]::get {Emmet/CSSPropertyEnd}]
1002  set separator [[ns preferences]::get {Emmet/CSSValueSeparator}]
1003  lappend prefix_list ""
1004  foreach prefix $prefix_list {
1005  lappend lines "$prefix$property$separator$values$endstr"
1006  }
1007  return [join $lines \n]
1008  } else {
1009  return [join $line " "]
1010  }
1011 
1012  }

1027  # Returns the prefix_list for the given property.
1028  proc get_prefix_list {property hyphen} {
1029 
1030  variable vendor_props
1031  variable prefixes
1032 
1033  set prefix_list [list]
1034 
1035  if {[[ns preferences]::get {Emmet/CSSAutoInsertVendorPrefixes}] || $hyphen} {
1036 
1037  # Get the override values from preferences
1038  set overrides(w) [[ns preferences]::get {Emmet/CSSWebkitPropertiesAddon}]
1039  set overrides(m) [[ns preferences]::get {Emmet/CSSMozPropertiesAddon}]
1040  set overrides(s) [[ns preferences]::get {Emmet/CSSMSPropertiesAddon}]
1041  set overrides(o) [[ns preferences]::get {Emmet/CSSOPropertiesAddon}]
1042 
1043  # If the value is in vender_props and its not removed via preferences
1044  if {[info exists vendor_props($property)]} {
1045  foreach prefix [split $vendor_props($property) {}] {
1046  if {[lsearch $overrides($prefix) -$property] == -1} {
1047  lappend prefix_list $prefixes($prefix)
1048  }
1049  }
1050 
1051  # Otherwise, check to see if the property was added by the user
1052  } else {
1053  foreach prefix [list w m s o] {
1054  if {[lsearch $overrides($prefix) $property] != -1} {
1055  lappend prefix_list $prefixes($prefix)
1056  }
1057  }
1058  }
1059 
1060  # If there are no matching prefixes, use them all
1061  if {([llength $prefix_list] == 0) && $hyphen} {
1062  foreach prefix [list w m s o] {
1063  lappend prefix_list $prefixes($prefix)
1064  }
1065  }
1066 
1067  }
1068 
1069  return $prefix_list
1070
