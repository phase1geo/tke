30  # This procedure is called by the sftp.tcl procedures. We will ignore
31  # their parameters.
32  proc ::frputs {args} {
33  return
34  set m {}
35  foreach ar $args {
36  if {[string index $ar end] == " " } {
37  set m [set m][string range $ar 0 end-1]
38  } elseif { ! [catch "uplevel \"info exists $ar\"" ro] && $ro } {
39  set m "[set m]$ar=[uplevel "set $ar"]< "
40  } else {
41  set m "[set m]$ar=<unset> "
42  }
43  }
44  regsub -all {\n} $m {\\n} m
45  regsub -all {\r} $m {\\r} m
46  regsub -all {\t} $m {\\t} m
47  puts "frputs: [set m]"
48  flush stdout
49  }
