27 # This procedure is called by the sftp.tcl procedures. We will ignore
28 # their parameters.
29 proc ::frputs {args} {
30  return
31  set m {}
32  foreach ar $args {
33  if {[string index $ar end] == " " } {
34  set m [set m][string range $ar 0 end-1]
35  } elseif { ! [catch "uplevel \"info exists $ar\"" ro] && $ro } {
36  set m "[set m]$ar=[uplevel "set $ar"]< "
37  } else {
38  set m "[set m]$ar=<unset> "
39  }
40  }
41  regsub -all {\n} $m {\\n} m
42  regsub -all {\r} $m {\\r} m
43  regsub -all {\t} $m {\\t} m
44  puts "frputs: [set m]"
45  flush stdout
46 }
