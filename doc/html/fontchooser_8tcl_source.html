215  # Called when one of the font variables are written to. Updates the UI.
216  proc tracer {var1 var2 op} {
217 
218  variable data
219 
220  lassign [split $var2 ,] w var
221 
222  # Clear the selection
223  $w.l${var}s selection clear 0 end
224 
225  # Find the exact (or closest) font match and get its index
226  set value [string tolower $data($w,$var)]
227  if {[set n [lsearch -exact $data($w,${var}s,lcase) $value]] == -1} {
228  if {[set n [lsearch -glob $data($w,${var}s,lcase) "$value*"]] == -1} {
229  return
230  }
231  }
232 
233  # Set the value
234  set data($w,$var) [lindex $data($w,${var}s) $n]
235 
236  # Update the UI
237  $w.e$var icursor end
238  $w.e$var selection clear
239  $w.l${var}s selection set $n
240  $w.l${var}s see $n
241 
242  # Display the font
243  show $w
244 
245  }

248  # Displays a sample of the selection options and generates the
249  # <<FontChanged>> virtual event.
250  proc show {w} {
251 
252  variable data
253 
254  set result [list -family $data($w,font) -size $data($w,size)]
255 
256  switch $data($w,style) {
257  "Bold" { lappend result -weight bold}
258  "Italic" { lappend result -slant italic}
259  "Bold Italic" { lappend result -weight bold -slant italic}
260  }
261 
262  if {$data($w,strike)} { lappend result -overstrike 1}
263  if {$data($w,under)} { lappend result -underline 1}
264 
265  # Display the sampled result and generate the FontChanged event
266  if {![catch { $data($w,sample) config -font $result}]} {
267  event generate $w <<FontChanged>> -data $result
268  }

273  # Returns the font families that match the given type.
274  proc find_font_class {{type mono}} {
275 
276  set fm [list]
277  set fv [list]
278  foreach f [font families] {
279  if {[font measure "{$f} 8" "A"] == [font measure "{$f} 8" "."]} {
280  lappend fm $f
281  } else {
282  lappend fv $f
283  }
284  }
285 
286  return [expr {($type eq "mono") ? $fm : $fv}]
287 
288  }
