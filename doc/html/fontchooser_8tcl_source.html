213  # Called when one of the font variables are written to. Updates the UI.
214  proc tracer {var1 var2 op} {
215 
216  variable data
217 
218  lassign [split $var2 ,] w var
219 
220  set bad 0
221 
222  # Make selection in each listbox
223  set value [string tolower $data($w,$var)]
224  $w.l${var}s selection clear 0 end
225  set n [lsearch -exact $data($w,${var}s,lcase) $value]
226  $w.l${var}s selection set $n
227  if {$n != -1} {
228  set data($w,$var) [lindex $data($w,${var}s) $n]
229  $w.e$var icursor end
230  $w.e$var selection clear
231  } else { ;# No match, try prefix
232  # Size is weird: valid numbers are legal but don't display
233  # unless in the font size list
234  set n [lsearch -glob $data($w,${var}s,lcase) "$value*"]
235  set bad 1
236  }
237  $w.l${var}s see $n
238 
239  if {!$bad} {
240  show $w
241  }
242

246  # Displays a sample of the selection options and generates the

248  proc show {w} {
249 
250  variable data
251 
252  set result [list -family $data($w,font) -size $data($w,size)]
253 
254  switch $data($w,style) {
255  "Bold" { lappend result -weight bold}
256  "Italic" { lappend result -slant italic}
257  "Bold Italic" { lappend result -weight bold -slant italic}

262

264  if {![catch { $data($w,sample) config -font $result}]} {

266  }

271  # Returns the font families that match the given type.
272  proc find_font_class {{type mono}} {
273 
274  set fm [list]
275  set fv [list]
276  foreach f [font families] {
277  if {[font measure "{$f} 8" "A"] == [font measure "{$f} 8" "."]} {
278  lappend fm $f
279  } else {
280  lappend fv $f
281  }
282  }
283 
284  return [expr {($type eq "mono") ? $fm : $fv}]
285 
286  }
